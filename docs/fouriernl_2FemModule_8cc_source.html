59/**
60 * @brief Performs the main computation for the FemModuleFourierNL.
61 *
62 * - Stops the time loop after 1 iteration since the equation is steady state.
63 * - Resets, configures, and initializes the linear system.
64 * - Executes the stationary solve.

67
68void FemModuleFourierNL::
69compute()
70{
71 info() << "[ArcaneFem-Info] Started module compute()";
72 Real elapsedTime = platform::getRealTime();
73
74 // Stop code after computations
75 if (m_global_iteration() > 0)
76 subDomain()->timeLoopMng()->stopComputeLoop(true);
77
78 m_linear_system.reset();
79 m_linear_system.setLinearSystemFactory(options()->linearSystem());
80 m_linear_system.initialize(subDomain(), acceleratorMng()->defaultRunner(), m_dofs_on_nodes.dofFamily(), "Solver");
81
82 if (m_petsc_flags != NULL){
83 CommandLineArguments args = ArcaneFemFunctions::GeneralFunctions::getPetscFlagsFromCommandline(m_petsc_flags);
84 m_linear_system.setSolverCommandLineArguments(args);
85 }
86
87 if (m_matrix_format == "BSR" || m_matrix_format == "AF-BSR") {
88 bool use_csr_in_linear_system =
89 options()->linearSystem.serviceName() == "HypreLinearSystem" ||
90 options()->linearSystem.serviceName() == "AlienLinearSystem" ||
91 options()->linearSystem.serviceName() == "PetscLinearSystem";
92 if (m_matrix_format == "BSR")
93 m_bsr_format.initialize(mesh(), 1, use_csr_in_linear_system, 0);
94 else
95 m_bsr_format.initialize(mesh(), 1, use_csr_in_linear_system, 1);
96 m_bsr_format.computeSparsity();
97 }
98
99 _doStationarySolve();
100
101 elapsedTime = platform::getRealTime() - elapsedTime;
102 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(), "compute", elapsedTime);
103}

105/*---------------------------------------------------------------------------*/
106/**
107 * @brief Performs a stationary solve for the FEM system.
108 *
109 * This method follows via the following steps:
110 * 1. _getMaterialParameters() Retrieves material parameters
111 * 2. _assembleBilinearOperator() Assembles the FEM matrix 𝐀
112 * 3. _assembleLinearOperator() Assembles the FEM RHS vector 𝐛
113 * 4. _solve() Solves for solution vector 𝐮 = 𝐀⁻¹𝐛
114 * 5. _updateVariables() Updates FEM variables 𝐮 = 𝐱
115 * 6. _validateResults() Regression test
116 */
117/*---------------------------------------------------------------------------*/
118
119void FemModuleFourierNL::
120_doStationarySolve()
121{
122 info() << "[ArcaneFem-Info] Started module _doStationarySolve()";
123 _updatePreviousIterationVariables();
124 while (m_fp_iter < m_max_fp_iters) {
125 if (m_assemble_linear_system) {
126
127 if (m_linear_system.isInitialized() && m_fp_iter != 0) {
128 m_linear_system.clearValues();
129
130 if (m_matrix_format == "BSR" || m_matrix_format == "AF-BSR")
131 m_bsr_format.resetMatrixValues();
132
133 _assembleBilinearOperator();
134
135 /* TODO : We should ideally not update the matrix row/columns concerning Dirichlet if Dirichlet BC are fixed for all iterations */
136 /* need to create a separate function _assembleLinearOperatorLHSOnly. */
137 _assembleLinearOperator(); // We use _assembleLinearOperator now for simplicity
138 }
139 else {
140 _assembleBilinearOperator();
141 _assembleLinearOperator();
142 }
143 }
144 if (m_solve_linear_system) {
145 _solve();
146 _updateVariables();
147 }
148
149 ++m_fp_iter;
150 _checkConvergence();
151
152 if (m_converged) {
153 _updateExactSolution();
154 info() << "[ArcaneFem-Info] Fixed-point iterations converged after " << m_fp_iter << " iterations";
155 break;
156 }
157 else {
158 _updatePreviousIterationVariables(); // copy u into uk for next iteration convergence check
159 // m_uk.copy(m_u);
160 _updateSolutionFromVariables(); // copy u into u_dof to update initial guess for linear solve TODO See how to use swap instead of deep copy
161 }
162 }
163 if (m_fp_iter == m_max_fp_iters && !m_converged) {
164 info() << "[ArcaneFem-Info] Fixed-point iterations did not converge after maximum (" << m_max_fp_iters << ") iterations";
165 ARCANE_FATAL("Fixed-point iterations diverged after max iters");
166 }
167 if (m_cross_validation) {
168 _validateResults();
169 }
170 if (m_check_solution) {
171 _checkSolution();
172 }
173}

185/*---------------------------------------------------------------------------*/
186/**
187 * @brief FEM linear operator for the current simulation step.
188 *
189 * This method constructs the linear system by assembling the LHS matrix
190 * and RHS vector, applying various boundary conditions and source terms.
191 *
192 * Steps involved:
193 * 1. The RHS vector is initialized to zero before applying any conditions.
194 * 2. If a constant source term is specified (`qdot`), apply it to the RHS.
195 * 3. If Neumann BC are specified applied to the RHS.
196 * 4. If Dirichlet BC are specified apply to the LHS & RHS.
197 */
198/*---------------------------------------------------------------------------*/
199
200void FemModuleFourierNL::
201_assembleLinearOperatorCpu()
202{
203 info() << "[ArcaneFem-Info] Started module _assembleLinearOperator()";
204 Real elapsedTime = platform::getRealTime();
205
206 if (m_matrix_format == "BSR" || m_matrix_format == "AF-BSR")
207 m_bsr_format.toLinearSystem(m_linear_system);
208
209 VariableDoFReal& rhs_values(m_linear_system.rhsVariable()); // Temporary variable to keep values for the RHS
210 rhs_values.fill(0.0);
211
212 auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
213
214 if (options()->qdot.isPresent()) {
215 if (mesh()->dimension() == 2) {
216 if (m_hex_quad_mesh)
217 ArcaneFemFunctions::BoundaryConditions2D::applyConstantSourceToRhsQuad4(m_qdot, mesh(), node_dof, m_node_coord, rhs_values);
218 else
219 ArcaneFemFunctions::BoundaryConditions2D::applyConstantSourceToRhs(m_qdot, mesh(), node_dof, m_node_coord, rhs_values);
220 }
221 else {
222 if (m_hex_quad_mesh)
223 ArcaneFemFunctions::BoundaryConditions3D::applyConstantSourceToRhsHexa8(m_qdot, mesh(), node_dof, m_node_coord, rhs_values);
224 else
225 ArcaneFemFunctions::BoundaryConditions3D::applyConstantSourceToRhs(m_qdot, mesh(), node_dof, m_node_coord, rhs_values);
226 }
227 }
228
229 // Helper lambda to apply boundary conditions
230 auto applyBoundaryConditions = [&](auto BCFunctions) {
231 BC::IArcaneFemBC* bc = options()->boundaryConditions();
232 if (bc) {
233
234 // Neumann
235 for (BC::INeumannBoundaryCondition* bs : bc->neumannBoundaryConditions()) {
236 if (mesh()->dimension() == 2) {
237 if (m_hex_quad_mesh)
238 ArcaneFemFunctions::BoundaryConditions2D::applyNeumannToRhsQuad4(bs, node_dof, m_node_coord, rhs_values);
239 else
240 ArcaneFemFunctions::BoundaryConditions2D::applyNeumannToRhs(bs, node_dof, m_node_coord, rhs_values);
241 }
242 if (mesh()->dimension() == 3) {
243 if (m_hex_quad_mesh)
244 ArcaneFemFunctions::BoundaryConditions3D::applyNeumannToRhsHexa8(bs, node_dof, m_node_coord, rhs_values);
245 else
246 ArcaneFemFunctions::BoundaryConditions3D::applyNeumannToRhs(bs, node_dof, m_node_coord, rhs_values);
247 }
248 }
249
250 // Dirichlet
251 for (BC::IDirichletBoundaryCondition* bs : bc->dirichletBoundaryConditions())
252 ArcaneFemFunctions::BoundaryConditions::applyDirichletToLhsAndRhs(bs, node_dof, m_linear_system, rhs_values);
253 }
254 };
255
256 if (mesh()->dimension() == 3)
257 applyBoundaryConditions(ArcaneFemFunctions::BoundaryConditions3D());
258 else
259 applyBoundaryConditions(ArcaneFemFunctions::BoundaryConditions2D());
260
261 elapsedTime = platform::getRealTime() - elapsedTime;
262 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(), "rhs-vector-assembly", elapsedTime);
263}

265/*---------------------------------------------------------------------------*/
266/**
267 * @brief FEM linear operator for the current simulation step.
268 * GPU compatible. Currently working with HypreDoFLinearSystem.
269 *
270 * This method constructs the linear system by assembling the LHS matrix
271 * and RHS vector, applying various boundary conditions and source terms.
272 *
273 * Steps involved:
274 * 1. The RHS vector is initialized to zero before applying any conditions.
275 * 2. If Neumann BC are specified applied to the RHS.
276 * 3. If Dirichlet BC/Point are specified apply to the LHS & RHS.
277 */
278/*---------------------------------------------------------------------------*/
279
280void FemModuleFourierNL::_assembleLinearOperatorGpu()
281{
282 info() << "[ArcaneFem-Info] Started module _assembleLinearOperatorGpu()";
283 Real elapsedTime = platform::getRealTime();
284
285 m_bsr_format.toLinearSystem(m_linear_system);
286
287 auto& rhs_values(m_linear_system.rhsVariable());
288 rhs_values.fill(0.0);
289
290 // Because of Dirichlet (Penalty) implementation in Hypre.
291 m_linear_system.getForcedInfo().fill(false);
292
293 auto queue = subDomain()->acceleratorMng()->defaultQueue();
294 auto mesh_ptr = mesh();
295
296 auto applyBoundaryConditions = [&](auto BCFunctions) {
297 if (options()->qdot.isPresent())
298 BCFunctions.applyConstantSourceToRhs(m_qdot, m_dofs_on_nodes, m_node_coord, rhs_values, mesh_ptr, queue);
299
300 BC::IArcaneFemBC* bc = options()->boundaryConditions();
301 if (bc) {
302 for (BC::INeumannBoundaryCondition* bs : bc->neumannBoundaryConditions())
303 BCFunctions.applyNeumannToRhs(bs, m_dofs_on_nodes, m_node_coord, rhs_values, mesh_ptr, queue);
304
305 for (BC::IDirichletBoundaryCondition* bs : bc->dirichletBoundaryConditions())
306 FemUtils::Gpu::BoundaryConditions::applyDirichletToLhsAndRhs(bs, m_dofs_on_nodes, m_linear_system, mesh_ptr, queue);
307
308 }
309 };
310
311 if (mesh()->dimension() == 3)
312 applyBoundaryConditions(FemUtils::Gpu::BoundaryConditions3D());
313 else
314 applyBoundaryConditions(FemUtils::Gpu::BoundaryConditions2D());
315
316 elapsedTime = platform::getRealTime() - elapsedTime;
317 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(),"rhs-vector-assembly-gpu", elapsedTime);
318}

326void FemModuleFourierNL::
327_assembleBilinearOperator()
328{
329 info() << "[ArcaneFem-Info] Started module _assembleBilinearOperator()";
330 Real elapsedTime = platform::getRealTime();
331
332 if (m_matrix_format == "BSR" || m_matrix_format == "AF-BSR") {
333 UnstructuredMeshConnectivityView m_connectivity_view(mesh());
334 auto cn_cv = m_connectivity_view.cellNode();
335 auto m_queue = subDomain()->acceleratorMng()->defaultQueue();
336 auto command = makeCommand(m_queue);
337 auto in_node_coord = ax::viewIn(command, m_node_coord);
338 auto in_node_uk = ax::viewIn(command, m_uk);
339
340 if (mesh()->dimension() == 2)
341 if (m_matrix_format == "BSR")
342 m_bsr_format.assembleBilinearAtomic([=] ARCCORE_HOST_DEVICE(CellLocalId cell_lid) { return computeElementMatrixTria3Gpu(cell_lid, cn_cv, in_node_coord, in_node_uk); });
343 else
344 m_bsr_format.assembleBilinearAtomicFree([=] ARCCORE_HOST_DEVICE(CellLocalId cell_lid, Int32 node_lid) { return computeElementVectorTria3Gpu(cell_lid, cn_cv, in_node_coord, in_node_uk, node_lid); });
345 else
346 if (m_matrix_format == "BSR")
347 m_bsr_format.assembleBilinearAtomic([=] ARCCORE_HOST_DEVICE(CellLocalId cell_lid) { return computeElementMatrixTetra4Gpu(cell_lid, cn_cv, in_node_coord, in_node_uk); });
348 else
349 m_bsr_format.assembleBilinearAtomicFree([=] ARCCORE_HOST_DEVICE(CellLocalId cell_lid, Int32 node_lid) { return computeElementVectorTetra4Gpu(cell_lid, cn_cv, in_node_coord, in_node_uk, node_lid); });
350 }
351
352 if (m_matrix_format == "DOK") {
353 if (mesh()->dimension() == 3)
354 if(m_hex_quad_mesh)
355 _assembleBilinear<8>([this](const Cell& cell) { return _computeElementMatrixHexa8(cell); });
356 else
357 _assembleBilinear<4>([this](const Cell& cell) { return _computeElementMatrixTetra4(cell); });
358
359 if (mesh()->dimension() == 2)
360 if(m_hex_quad_mesh)
361 _assembleBilinear<4>([this](const Cell& cell) { return _computeElementMatrixQuad4(cell); });
362 else
363 _assembleBilinear<3>([this](const Cell& cell) { return _computeElementMatrixTria3(cell); });
364 }
365
366 elapsedTime = platform::getRealTime() - elapsedTime;
367 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(), "lhs-matrix-assembly", elapsedTime);
368}

370/*---------------------------------------------------------------------------*/
371/**
372 * @brief Assembles the bilinear operator matrix for the FEM linear system.
373 *
374 * The method performs the following steps:
375 * 1. Computes element matrix using provided `compute_element_matrix` function.
376 * 2. Assembles global matrix by adding contributions from each cell's element
377 * matrix to the corresponding entries in the global matrix.
378 */
379/*---------------------------------------------------------------------------*/
380
381template <int N>
382void FemModuleFourierNL::
383_assembleBilinear(const std::function<RealMatrix<N, N>(const Cell&)>& compute_element_matrix)
384{
385 auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
386
387 ENUMERATE_ (Cell, icell, allCells()) {
388 Cell cell = *icell;
389
390 auto K_e = compute_element_matrix(cell); // element matrix based on the provided function
391 Int32 n1_index = 0;
392 for (Node node1 : cell.nodes()) {
393 Int32 n2_index = 0;
394 for (Node node2 : cell.nodes()) {
395 Real v = K_e(n1_index, n2_index);
396 if (node1.isOwn()) {
397 m_linear_system.matrixAddValue(node_dof.dofId(node1, 0), node_dof.dofId(node2, 0), v);
398 }
399 ++n2_index;
400 }
401 ++n1_index;
402 }
403 }
404}

431 * @brief Check for convergence and Update the FEM variables.
432 *
433 * This method performs the following actions:
434 * 1. Updates the FEM solutions from the solutions of linear solver on DOF

442_checkConvergence()
443{
444 info() << "[ArcaneFem-Info] Started module _checkConvergence()";
445 Real elapsedTime = platform::getRealTime();
446
447 m_u.synchronize();
448 m_uk.synchronize();
449
450 Real max_error = 0.0;
451 // Real l1_error = 0.0;
452 {
453 ENUMERATE_ (Node, inode, ownNodes()) {
454 const Real error = math::abs(m_u[inode] - m_uk[inode]);
455
456 max_error = math::max(error, max_error);
457 // l1_error += error;
458 }
459 }
460 IParallelMng* pm = defaultMesh()->parallelMng();
461 max_error = pm->reduce(Parallel::ReduceMax, max_error);
462 // l1_error = pm->reduce(Parallel::ReduceSum, l1_error);
463
464 // if ( max_error < m_fp_tol || l1_error < m_fp_tol){
465 if ( max_error < m_fp_tol){
466 // if ( l1_error < m_fp_tol ){
467 m_converged = true;
468 } else {
469 m_converged = false;
470 }
471
472 // info() << "[ArcaneFem-FP-iters] At fixed-point iteration "<< m_fp_iter <<": linf(max)-error = " << max_error << " and l1-error = " << l1_error;
473 info() << "[ArcaneFem-Info] At fixed-point iteration "<< m_fp_iter <<": linf(max)-error = " << max_error;
474 // info() << "[ArcaneFem-FP-iters] At fixed-point iteration "<< m_fp_iter <<": l1-error = " << l1_error;

633
634void FemModuleFourierNL::
635_updateExactSolution(bool verbose)
636{
637 info() << "[ArcaneFem-Info] Started module _updateExactSolution()";
638 Real elapsedTime = platform::getRealTime();
639
640 {
641 ENUMERATE_ (Node, inode, ownNodes()) {
642 Node node = *inode;
643 Real m = options()->expNlin;
644 m_u_exact[node] = math::pow((math::pow(2.0, m + 1) - 1) * m_node_coord[node].x + 1, 1 / (m + 1)) - 1.0;
645 if (verbose) {
646 info() << "u_exact[" << node.uniqueId() << "] = " << m_u_exact[node];
647 }
648 }
649 }
650 m_u_exact.synchronize();
651
652 elapsedTime = platform::getRealTime() - elapsedTime;
653 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(), "update-exact-solution", elapsedTime);
654}

672 Real elapsedTime = platform::getRealTime();
673
674 Real max_error = 0.0;
675 {
676 ENUMERATE_ (Node, inode, ownNodes()) {
677 const Real error = math::abs(m_u[inode] - m_u_exact[inode]);
678
679 max_error = math::max(error, max_error);
680 }
681 }
682 IParallelMng* pm = defaultMesh()->parallelMng();

687 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(), "solution-check", elapsedTime);
688}
