15/**
16 * @brief Computes the element matrix for a triangular element (ℙ1 FE).
17 *
18 * Theory:
19 *
20 * a(𝐮,𝐯) = ∫∫ σ(𝐮):ε(𝐯)dΩ with 𝐮 = (𝑢𝑥,𝑢𝑦) and 𝐯 = (𝑣𝑥,𝑣𝑦)
21 * σ(𝐮) is stress tensor with σᵢⱼ = λδᵢⱼεₖₖ + 2μεᵢⱼ
22 * ε(𝐯) is strain tensor with εᵢⱼ = 0.5 (∂𝑣ᵢ/∂xⱼ + ∂𝑣ⱼ/∂xᵢ)
23 *
24 * the bilinear integral expands to
25 *
26 * a(𝐮,𝐯) = ∫∫ [σ_𝑥𝑥ε_𝑥𝑥 + σ_𝑦𝑦ε_𝑦𝑦 + 2σ_𝑥𝑦ε_𝑥𝑦]dΩ
27 *
28 * this further expands to
29 *
30 * a(𝐮,𝐯) = ∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦)
31 * + ∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦)
32 * + ∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥)
33 *
34 * - The first term is "normal strain energy"
35 * - The second term is "compressibility effect"
36 * - The third term is "shear energy"
37 *
38 */
39/*---------------------------------------------------------------------------*/
40
41ARCCORE_HOST_DEVICE RealMatrix<6, 6> computeElementMatrixTria3Base(Real3 dxu, Real3 dyu, Real area, Real lambda, Real mu)
42{
43 RealVector<6> dxUx = { dxu[0], 0., dxu[1], 0., dxu[2], 0. };
44 RealVector<6> dyUx = { dyu[0], 0., dyu[1], 0., dyu[2], 0. };
45 RealVector<6> dxUy = { 0., dxu[0], 0., dxu[1], 0., dxu[2] };
46 RealVector<6> dyUy = { 0., dyu[0], 0., dyu[1], 0., dyu[2] };
47
48 // ∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦)
49 RealMatrix<6, 6> normal_strain_energy = (lambda + 2 * mu) * ((dxUx ^ dxUx) + (dyUy ^ dyUy)) * area;
50
51 // ∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦)
52 RealMatrix<6, 6> compressibility_effect = (lambda) * ((dyUy ^ dxUx) + (dxUx ^ dyUy)) * area;
53
54 // ∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥)
55 RealMatrix<6, 6> shear_energy = (mu) * ((dxUy + dyUx) ^ (dyUx + dxUy)) * area;
56
57 return (normal_strain_energy + compressibility_effect + shear_energy);
58}

87ARCCORE_HOST_DEVICE RealMatrix<2, 6> computeElementVectorTria3Gpu(CellLocalId cell_lid, const IndexedCellNodeConnectivityView& cn_cv, const Accelerator::VariableNodeReal3InView& in_node_coord, Real lambda, Real mu, Int32 node_lid)
88{
89 Real3 dxu = Arcane::FemUtils::Gpu::FeOperation2D::computeGradientXTria3(cell_lid, cn_cv, in_node_coord);
90 Real3 dyu = Arcane::FemUtils::Gpu::FeOperation2D::computeGradientYTria3(cell_lid, cn_cv, in_node_coord);
91 Real area = Arcane::FemUtils::Gpu::MeshOperation::computeAreaTria3(cell_lid, cn_cv, in_node_coord);
92
93 RealVector<6> dxUx = { dxu[0], 0., dxu[1], 0., dxu[2], 0. };
94 RealVector<6> dyUx = { dyu[0], 0., dyu[1], 0., dyu[2], 0. };
95 RealVector<6> dxUy = { 0., dxu[0], 0., dxu[1], 0., dxu[2] };
96 RealVector<6> dyUy = { 0., dyu[0], 0., dyu[1], 0., dyu[2] };
97
98 // ∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦)
99 RealVector<6> normal_strain_energy_x = (lambda + 2 * mu) * ((dxUx(node_lid*2) * dxUx) + (dyUy(node_lid*2) * dyUy)) * area;
100 RealVector<6> normal_strain_energy_y = (lambda + 2 * mu) * ((dxUx(node_lid*2+1) * dxUx) + (dyUy(node_lid*2+1) * dyUy)) * area;
101
102 // ∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦)
103 RealVector<6> compressibility_effect_x = (lambda) * ((dyUy(node_lid*2) * dxUx) + (dxUx(node_lid*2) * dyUy)) * area;
104 RealVector<6> compressibility_effect_y = (lambda) * ((dyUy(node_lid*2+1) * dxUx) + (dxUx(node_lid*2+1) * dyUy)) * area;
105
106 // ∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥)
107 RealVector<6> shear_energy_x = (mu) * ((dxUy(node_lid*2) + dyUx(node_lid*2)) * (dyUx + dxUy)) * area;
108 RealVector<6> shear_energy_y = (mu) * ((dxUy(node_lid*2+1) + dyUx(node_lid*2+1)) * (dyUx + dxUy)) * area;
109
110 RealVector <6> result_x = normal_strain_energy_x + compressibility_effect_x + shear_energy_x;
111 RealVector <6> result_y = normal_strain_energy_y + compressibility_effect_y + shear_energy_y;
112
113 RealMatrix<2, 6> result = {
114 { result_x(0), result_x(1), result_x(2), result_x(3), result_x(4), result_x(5) },
115 { result_y(0), result_y(1), result_y(2), result_y(3), result_y(4), result_y(5) }
116 };
117
118 return result;
119}

122/**
123 * @brief Computes the element matrix for a tetrahedral element (ℙ1 FE).
124 *
125 * Theory:
126 *
127 * a(𝐮,𝐯) = ∫∫∫ [σ(𝐮):ε(𝐯)dΩ with 𝐮 = (𝑢𝑥, 𝑢𝑦, 𝑢𝑧) and 𝐯 = (𝑣𝑥, 𝑣𝑦, 𝑣𝑧)
128 *
129 * where:
130 *
131 * σ(𝐮) is stress tensor with σᵢⱼ = λδᵢⱼεₖₖ + 2μεᵢⱼ
132 * ε(𝐯) is strain tensor with εᵢⱼ = 0.5 (∂𝑣ᵢ/∂xⱼ + ∂𝑣ⱼ/∂xᵢ)
133 *
134 * the bilinear integral expands to
135 *
136 * a(𝐮,𝐯) = ∫∫∫ [σ_𝑥𝑥ε_𝑥𝑥 + σ_𝑦𝑦ε_𝑦𝑦 + σ_𝑧𝑧ε_𝑧𝑧 + 2σ_𝑥𝑦ε_𝑥𝑦 + 2σ_𝑦𝑧ε_𝑦𝑧 + 2σ_𝑧𝑥ε_𝑧𝑥]dΩ
137 *
138 * this further expands to
139 *
140 * a(𝐮,𝐯) = ∫∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑧/∂𝑧)
141 * + ∫∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑧/∂𝑧 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑧/∂𝑧 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑦/∂𝑦 )
142 * + ∫∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥) + μ(∂𝑢𝑧/∂𝑦 + ∂𝑢𝑦/∂𝑧)(∂𝑣𝑦/∂𝑧 + ∂𝑣𝑧/∂𝑦) + μ(∂𝑢𝑥/∂𝑧 + ∂𝑢𝑧/∂𝑥)(∂𝑣𝑧/∂𝑥 + ∂𝑣𝑥/∂𝑧)
143 *
144 * - The first term is "normal strain energy"
145 * - The second term is "compressibility effect"
146 * - The third term is "shear energy"
147 *
148 */
149/*---------------------------------------------------------------------------*/
150
151ARCCORE_HOST_DEVICE RealMatrix<12, 12> computeElementMatrixTetra4Base(Real4 dxu, Real4 dyu, Real4 dzu, Real volume, Real lambda, Real mu)
152{
153 RealVector<12> dxUx = { dxu[0], 0., 0., dxu[1], 0., 0., dxu[2], 0., 0., dxu[3], 0., 0. };
154 RealVector<12> dyUx = { dyu[0], 0., 0., dyu[1], 0., 0., dyu[2], 0., 0., dyu[3], 0., 0. };
155 RealVector<12> dzUx = { dzu[0], 0., 0., dzu[1], 0., 0., dzu[2], 0., 0., dzu[3], 0., 0. };
156
157 RealVector<12> dxUy = { 0., dxu[0], 0., 0., dxu[1], 0., 0., dxu[2], 0., 0., dxu[3], 0. };
158 RealVector<12> dyUy = { 0., dyu[0], 0., 0., dyu[1], 0., 0., dyu[2], 0., 0., dyu[3], 0. };
159 RealVector<12> dzUy = { 0., dzu[0], 0., 0., dzu[1], 0., 0., dzu[2], 0., 0., dzu[3], 0. };
160
161 RealVector<12> dxUz = { 0., 0., dxu[0], 0., 0., dxu[1], 0., 0., dxu[2], 0., 0., dxu[3] };
162 RealVector<12> dyUz = { 0., 0., dyu[0], 0., 0., dyu[1], 0., 0., dyu[2], 0., 0., dyu[3] };
163 RealVector<12> dzUz = { 0., 0., dzu[0], 0., 0., dzu[1], 0., 0., dzu[2], 0., 0., dzu[3] };
164
165 // ∫∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑧/∂𝑧)
166 RealMatrix<12, 12> normal_strain_energy = (lambda + 2 * mu) * ((dxUx ^ dxUx) + (dyUy ^ dyUy) + (dzUz ^ dzUz) ) * volume;
167
168 // ∫∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦
169 // + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑧/∂𝑧
170 // + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑧/∂𝑧 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑦/∂𝑦)
171 RealMatrix<12, 12> compressibility_effect = (lambda) * ((dyUy ^ dxUx) + (dxUx ^ dyUy) +
172 (dzUz ^ dxUx) + (dxUx ^ dzUz) +
173 (dyUy ^ dzUz) + (dzUz ^ dyUy) ) * volume;
174
175 // ∫∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥) +
176 // μ(∂𝑢𝑧/∂𝑦 + ∂𝑢𝑦/∂𝑧)(∂𝑣𝑦/∂𝑧 + ∂𝑣𝑧/∂𝑦) +
177 // μ(∂𝑢𝑥/∂𝑧 + ∂𝑢𝑧/∂𝑥)(∂𝑣𝑧/∂𝑥 + ∂𝑣𝑥/∂𝑧)
178 RealMatrix<12, 12> shear_energy = (mu) * ( ((dxUy + dyUx) ^ (dyUx + dxUy)) +
179 ((dzUy + dyUz) ^ (dyUz + dzUy)) +
180 ((dxUz + dzUx) ^ (dzUx + dxUz)) ) * volume;
181
182 return (normal_strain_energy + compressibility_effect + shear_energy);
183}

216ARCCORE_HOST_DEVICE RealMatrix<3, 12> computeElementVectorTetra4Gpu(CellLocalId cell_lid, const IndexedCellNodeConnectivityView& cn_cv, const Accelerator::VariableNodeReal3InView& in_node_coord, Real lambda, Real mu, Int32 node_lid)
217{
218 Real4 dxu = Arcane::FemUtils::Gpu::FeOperation3D::computeGradientXTetra4(cell_lid, cn_cv, in_node_coord);
219 Real4 dyu = Arcane::FemUtils::Gpu::FeOperation3D::computeGradientYTetra4(cell_lid, cn_cv, in_node_coord);
220 Real4 dzu = Arcane::FemUtils::Gpu::FeOperation3D::computeGradientZTetra4(cell_lid, cn_cv, in_node_coord);
221
222 Real volume = Arcane::FemUtils::Gpu::MeshOperation::computeVolumeTetra4(cell_lid, cn_cv, in_node_coord);
223
224 RealVector<12> dxUx = { dxu[0], 0., 0., dxu[1], 0., 0., dxu[2], 0., 0., dxu[3], 0., 0. };
225 RealVector<12> dyUx = { dyu[0], 0., 0., dyu[1], 0., 0., dyu[2], 0., 0., dyu[3], 0., 0. };
226 RealVector<12> dzUx = { dzu[0], 0., 0., dzu[1], 0., 0., dzu[2], 0., 0., dzu[3], 0., 0. };
227
228 RealVector<12> dxUy = { 0., dxu[0], 0., 0., dxu[1], 0., 0., dxu[2], 0., 0., dxu[3], 0. };
229 RealVector<12> dyUy = { 0., dyu[0], 0., 0., dyu[1], 0., 0., dyu[2], 0., 0., dyu[3], 0. };
230 RealVector<12> dzUy = { 0., dzu[0], 0., 0., dzu[1], 0., 0., dzu[2], 0., 0., dzu[3], 0. };
231
232 RealVector<12> dxUz = { 0., 0., dxu[0], 0., 0., dxu[1], 0., 0., dxu[2], 0., 0., dxu[3] };
233 RealVector<12> dyUz = { 0., 0., dyu[0], 0., 0., dyu[1], 0., 0., dyu[2], 0., 0., dyu[3] };
234 RealVector<12> dzUz = { 0., 0., dzu[0], 0., 0., dzu[1], 0., 0., dzu[2], 0., 0., dzu[3] };
235
236 // ∫∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 +
237 // ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦 +
238 // ∂𝑢𝑧/∂𝑧 ∂𝑣𝑧/∂𝑧 )
239 RealVector<12> normal_strain_energy_x = (lambda + 2 * mu) * ( (dxUx(node_lid*3) * dxUx) +
240 (dyUy(node_lid*3) * dyUy) +
241 (dzUz(node_lid*3) * dzUz) ) * volume;
242
243 RealVector<12> normal_strain_energy_y = (lambda + 2 * mu) * ( (dxUx(node_lid*3+1) * dxUx) +
244 (dyUy(node_lid*3+1) * dyUy) +
245 (dzUz(node_lid*3+1) * dzUz) ) * volume;
246
247 RealVector<12> normal_strain_energy_z = (lambda + 2 * mu) * ( (dxUx(node_lid*3+2) * dxUx) +
248 (dyUy(node_lid*3+2) * dyUy) +
249 (dzUz(node_lid*3+2) * dzUz) ) * volume;
250
251 // ∫∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦
252 // + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑧/∂𝑧
253 // + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑧/∂𝑧 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑦/∂𝑦)
254 RealVector<12> compressibility_effect_x = (lambda) * ((dyUy(node_lid*3) * dxUx) + (dxUx(node_lid*3) * dyUy) +
255 (dzUz(node_lid*3) * dxUx) + (dxUx(node_lid*3) * dzUz) +
256 (dyUy(node_lid*3) * dzUz) + (dzUz(node_lid*3) * dyUy) ) * volume;
257
258 RealVector<12> compressibility_effect_y = (lambda) * ((dyUy(node_lid*3+1) * dxUx) + (dxUx(node_lid*3+1) * dyUy) +
259 (dzUz(node_lid*3+1) * dxUx) + (dxUx(node_lid*3+1) * dzUz) +
260 (dyUy(node_lid*3+1) * dzUz) + (dzUz(node_lid*3+1) * dyUy) ) * volume;
261
262 RealVector<12> compressibility_effect_z = (lambda) * ((dyUy(node_lid*3+2) * dxUx) + (dxUx(node_lid*3+2) * dyUy) +
263 (dzUz(node_lid*3+2) * dxUx) + (dxUx(node_lid*3+2) * dzUz) +
264 (dyUy(node_lid*3+2) * dzUz) + (dzUz(node_lid*3+2) * dyUy) ) * volume;
265
266
267 // ∫∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥) +
268 // μ(∂𝑢𝑧/∂𝑦 + ∂𝑢𝑦/∂𝑧)(∂𝑣𝑦/∂𝑧 + ∂𝑣𝑧/∂𝑦) +
