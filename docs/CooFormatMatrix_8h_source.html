233 /**
234 * @brief Quicksort algorithm for the CSR Matrix
235 *
236 * @param is_row
237 * @param start
238 * @param end
239 */
240 void
241 sortMatrix(bool is_row, Int32 start, Int32 end)
242 {
243 if (start >= end) {
244 return;
245 }
246
247 int pivot = partition(is_row, start, end);
248
249 sortMatrix(is_row, start, pivot - 1);
250
251 sortMatrix(is_row, pivot + 1, end);
252 }

254 /**
255 * @brief Partition helper for the quickSort
256 *
257 * @param is_row
258 * @param start
259 * @param end
260 * @return Int32
261 */
262 Int32 partition(bool is_row, Int32 start, Int32 end)
263 {
264 Int32 pivot;
265 if (is_row)
266 pivot = m_matrix_row[end];
267 else
268 pivot = m_matrix_column[end];
269
270 Int32 pIndex = start;
271
272 for (Int32 i = start; i < end; i++) {
273 if ((is_row && m_matrix_row[i] <= pivot) || (!is_row && m_matrix_column[i] <= pivot)) {
274
275 swap(is_row, i, pIndex);
276 pIndex++;
277 }
278 }
279
280 swap(is_row, pIndex, end);
281
282 return pIndex;
283 }

285 /**
286 * @brief Swap helper for the quickSort
287 *
288 * @param is_row
289 * @param i
290 * @param j
291 */
292 void swap(bool is_row, Int32 i, Int32 j)
293 {
294 if (is_row) {
295 Int32 tmp = m_matrix_row(i);
296 m_matrix_row(i) = m_matrix_row(j);
297 m_matrix_row(j) = tmp;
298 }
299 Int32 tmp = m_matrix_column(i);
300 m_matrix_column(i) = m_matrix_column(j);
301 m_matrix_column(j) = tmp;
302 Real tmp_val = m_matrix_value(i);
303 m_matrix_value(i) = m_matrix_value(j);
304 m_matrix_value(j) = tmp_val;
305 }

308ARCCORE_HOST_DEVICE Int32 findIndexBinarySearch(Int32 row, Int32 col,
309 const auto& in_row_coo,
310 const auto& in_col_coo,
311 Int32 row_length)
312{
313 Int32 begin = 0;
314 Int32 end = row_length - 1;
315 Int32 i = -1; // Default value in case the index is not found
316
317 // Binary search to find the first occurrence of `row`
318 while (begin <= end) {
319 Int32 mid = begin + (end - begin) / 2;
320
321 if (row == in_row_coo(mid)) {
322 // Move back to the first occurrence of the row
323 while (mid > 0 && in_row_coo(mid - 1) == row) {
324 mid--;
325 }
326 i = mid;
327 break;
328 }
329
330 if (row > in_row_coo(mid)) {
331 begin = mid + 1;
332 }
333 else {
334 end = mid - 1;
335 }
336 }
337
338 // If no occurrence of the row was found, return -1
339 if (i == -1) {
340 return -1;
341 }
342
343 // Search for the matching column in the found row's block
344 while (i < row_length && in_row_coo(i) == row) {
345 if (in_col_coo(i) == col) {
346 return i;
347 }
348 i++;
349 }
350
351 // If no matching column is found, return -1
352 return -1;
353}
