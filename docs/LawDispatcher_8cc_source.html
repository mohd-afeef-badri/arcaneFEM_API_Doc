55LawDispatcher::LawDispatcher(TypesNLDynamic::eLawType law_type, bool default_param) :
56 m_law_type(law_type), m_default(default_param) {
57
58 // Setting to null default value
59 for (Integer i = 0; i < NB_LAW_TYPE; ++i) {
60 m_compute_stress[i] = nullptr;
61 m_compute_elast_tensor[i] = nullptr;
62 m_compute_tangent_tensor[i] = nullptr;
63 m_init_state[i] = nullptr;
64 m_init_history_vars[i] = nullptr;
65 m_read_law_params[i] = nullptr;
66 m_init_consts[i] = nullptr;
67 }
68
69 //! Computes stresses allowing to the material constitutive model type
70 m_compute_stress[TypesNLDynamic::HOOKE] = HookeComputeStress;
71 m_compute_stress[TypesNLDynamic::DRUCKP] = DruckPComputeStress;
72
73 //! Computes elastic constitutive tensor allowing to the material constitutive model type
74 m_compute_elast_tensor[TypesNLDynamic::HOOKE] = HookeComputeElastTensor;
75 m_compute_elast_tensor[TypesNLDynamic::DRUCKP] = DruckPComputeElastTensor;
76
77 //! Computes the tangent constitutive tensor allowing to the material constitutive model type (used for nonlinear models only)
78 m_compute_tangent_tensor[TypesNLDynamic::HOOKE] = HookeComputeTangentTensor;
79 m_compute_tangent_tensor[TypesNLDynamic::DRUCKP] = DruckPComputeTangentTensor;
80
81 //! Initializes the state allowing to the material constitutive model type (used for nonlinear models only)
82 m_init_state[TypesNLDynamic::HOOKE] = HookeInitState;
83 m_init_state[TypesNLDynamic::DRUCKP] = DruckPInitState;
84
85 //! Initializes the vector of intern (history) variables allowing to the material constitutive model type (used for nonlinear models only)
86 m_init_history_vars[TypesNLDynamic::HOOKE] = HookeInitHistoryVars;
87 m_init_history_vars[TypesNLDynamic::DRUCKP] = DruckPInitHistoryVars;
88
89 //! Read constitutive parameters from a file and initialize intern constants allowing to the material constitutive model type
90 m_read_law_params[TypesNLDynamic::HOOKE] = HookeReadLawParams;
91 m_read_law_params[TypesNLDynamic::DRUCKP] = DruckPReadLawParams;
92
93 //! Initialize intern useful constants allowing to the material constitutive model type
94 m_init_consts[TypesNLDynamic::HOOKE] = HookeInitConsts;
95 m_init_consts[TypesNLDynamic::DRUCKP] = DruckPInitConsts;
96
97 switch(law_type){
98
99 case TypesNLDynamic::HOOKE: m_nb_law_param = 2; m_nb_law_history_param = 0;
100 break;
101
102 case TypesNLDynamic::DRUCKP:
103 case TypesNLDynamic::MOHRC: m_nb_law_param = 7; m_nb_law_history_param = 1;
104 break;
105
106 case TypesNLDynamic::UNKNOWN:
107 default: m_nb_law_param = 2; m_nb_law_history_param = 0;
108 break;
109 }
110}

132void LawDispatcher::computeStress(bool init, bool isRef) {
133
134 auto f = m_compute_stress[m_law_type];
135 RealConstArrayView law_params = m_law_params.constView();
136 RealArrayView history_vars = m_history_vars.view();
137
138 if (f != nullptr)
139 {
140 Tensor4 tangent_tensor = f(law_params,history_vars,m_sig,m_eps,m_epsp,m_dsig,m_deps,isRef);
141 if (init || isRef) m_tangent_tensor = tangent_tensor;
142 }
143}

174bool LawDispatcher::initState(const Tensor2& sig)
175{
176 if (sig == Tensor2::zero()) return true;
177
178 ConstArrayView<Real> history_vars = m_history_vars.constView();
179 m_history_vars = initHistoryVars(history_vars);
180
181 auto f = m_init_state[m_law_type];
182 RealArrayView histab = m_history_vars.view();
183
184 if (f != nullptr)
185 return f(sig,histab);
186
187 return false;
188}

211RealUniqueArray LawDispatcher::readLawParams(Real lambda, Real mu, bool default_param, const String& name, Integer ilaw)
212{
213 auto f = m_read_law_params[m_law_type];
214 m_default = default_param;
215
216 if (f != nullptr) {
217 m_law_params = f(lambda, mu, default_param, name, ilaw);
218
219 if (!m_law_params.empty()) {
220 RealConstArrayView lawparams = m_law_params.constView();
221 m_law_consts = initConsts(lawparams);
222 }
223 }
224 else{
225 m_law_params = RealUniqueArray(2);
226 m_law_params[0] = lambda;
227 m_law_params[1] = mu;
228 }
229 return m_law_params;
230}

246void ReadLawBlock(istream& is, Integer nblock) {
247
248 char c[500];
249 String str;
250 bool stop{ false };
251 String str_nblock = String::fromNumber(nblock);
252 String sblock = String("<") + str_nblock + String(">");
253
254 do {
255
256 is.getline(c, 500); // read whole line
257 std::istringstream iss(c);
258 String token;
259 while (iss >> token) {
260 if (token.contains(sblock))
261 stop = true;
262 }
263 } while(!stop);
264}
