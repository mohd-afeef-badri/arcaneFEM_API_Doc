15/**
16 * @brief Applies double-couple boundary conditions to the RHS values.
17 *
18 * This function iterates over the defined double-couple boundary conditions,
19 * retrieves the corresponding force values from the associated CaseTable,
20 * and updates the RHS values for the specified nodes accordingly.
21 *
22 * @param rhs_values The RHS values to be updated.
23 * @param node_dof The node DoF connectivity view.
24 *
25/*---------------------------------------------------------------------------*/
26
27void FemModuleSoildynamics::
28_applyDoubleCouple(VariableDoFReal& rhs_values, const IndexedNodeDoFConnectivityView& node_dof)
29{
30 // Index of the boundary condition. Needed to associate a CaseTable
31 Int32 boundary_condition_index = 0;
32
33 const Int8 NorthSouthId = 0; // X-Direction DoF imposed by double-couple
34 const Int8 EastWestId = (mesh()->dimension() == 3) ? 2 : 1; // Y or Z-Direction DoF imposed by double-couple
35
36 for (const auto& bs : options()->doubleCouple()) {
37
38 const CaseTableInfo& case_table_dc_info = m_double_couple_case_table_list[boundary_condition_index];
39
40 ++boundary_condition_index;
41
42 Real dc_force; // double-couple force
43
44 CaseTable* dc_case_table = case_table_dc_info.case_table;
45 dc_case_table->value(t, dc_force);
46
47 NodeGroup north = bs->northNodeName();
48 NodeGroup south = bs->southNodeName();
49 NodeGroup east = bs->eastNodeName();
50 NodeGroup west = bs->westNodeName();
51
52 ENUMERATE_ (Node, inode, north.own()) {
53 Node node = *inode;
54 DoFLocalId dof_id1 = node_dof.dofId(node, NorthSouthId);
55 rhs_values[dof_id1] = dc_force;
56 }
57 ENUMERATE_ (Node, inode, south.own()) {
58 Node node = *inode;
59 DoFLocalId dof_id1 = node_dof.dofId(node, NorthSouthId);
60 rhs_values[dof_id1] = -dc_force;
61 }
62 ENUMERATE_ (Node, inode, east.own()) {
63 Node node = *inode;
64 DoFLocalId dof_id2 = node_dof.dofId(node, EastWestId);
65 rhs_values[dof_id2] = -dc_force;
66 }
67 ENUMERATE_ (Node, inode, west.own()) {
68 Node node = *inode;
69 DoFLocalId dof_id2 = node_dof.dofId(node, EastWestId);
70 rhs_values[dof_id2] = dc_force;
71 }
72 }
73}
