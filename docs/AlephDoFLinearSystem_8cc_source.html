261void AlephDoFLinearSystemImpl::
262_fillRowColumnEliminationInfos()
263{
264 OrderedRowColumnMap& rc_elimination_map = _rowColumnEliminationMap();
265 rc_elimination_map.clear();
266 DoFInfoListView item_list_view(dofFamily());
267
268 auto& dof_elimination_info = getEliminationInfo();
269 auto& dof_elimination_value = getEliminationValue();
270
271 for (const auto& rc_value : m_values_map) {
272 RowColumn rc = rc_value.first;
273 Real value = rc_value.second;
274 DoF dof_row = item_list_view[rc.row_id];
275 DoF dof_column = item_list_view[rc.column_id];
276 Byte row_elimination_info = dof_elimination_info[dof_row];
277 Byte column_elimination_info = dof_elimination_info[dof_column];
278 if (row_elimination_info == ELIMINATE_ROW_COLUMN || column_elimination_info == ELIMINATE_ROW_COLUMN)
279 rc_elimination_map[{ rc.row_id, rc.column_id }] = value;
280 }
281}

286void AlephDoFLinearSystemImpl::
287_applyMatrixTransformationAndFillAlephMatrix()
288{
289 _fillRowColumnEliminationInfos();
290 OrderedRowColumnMap& rc_elimination_map = _rowColumnEliminationMap();
291
292 IItemFamily* dof_family = dofFamily();
293 DoFInfoListView item_list_view(dof_family);
294
295 auto& dof_elimination_info = getEliminationInfo();
296 auto& dof_elimination_value = getEliminationValue();
297
298 // Fill the matrix from the values of \a m_values_map
299 // Skip (row,column) values which are part of an elimination.
300 for (const auto& rc_value : m_values_map) {
301 RowColumn rc = rc_value.first;
302 Real value = rc_value.second;
303 DoF dof_row = item_list_view[rc.row_id];
304 DoF dof_column = item_list_view[rc.column_id];
305
306 Byte row_elimination_info = dof_elimination_info[dof_row];
307
308 if (row_elimination_info == ELIMINATE_ROW)
309 // Will be computed after this loop
310 continue;
311 if (rc_elimination_map.contains(rc))
312 continue;
313
314 // Check if value is forced for current RowColumn
315 auto x = m_forced_set_values_map.find(rc);
316 if (x != m_forced_set_values_map.end()) {
317 info(4) << "FORCED VALUE R=" << rc.row_id << " C=" << rc.column_id
318 << " old=" << value << " new=" << x->second;
319 value = x->second;
320 }
321
322 _setMatrixValue(dof_row, dof_column, value);
323 }
324
325 const bool do_print_filling = m_do_print_filling;
326
327 // Apply Row or Row+Column elimination on Matrix
328 // Phase 2: set the diagonal value for elimination row to 1.0
329 ENUMERATE_ (DoF, idof, dof_family->allItems()) {
330 DoF dof = *idof;
331 if (!dof.isOwn())
332 continue;
333 Byte elimination_info = dof_elimination_info[dof];
334 if (elimination_info == ELIMINATE_ROW || elimination_info == ELIMINATE_ROW_COLUMN) {
335 Real elimination_value = dof_elimination_value[dof];
336 if (do_print_filling)
337 info() << "EliminateMatrix info=" << static_cast<int>(elimination_info) << " row="
338 << std::setw(4) << dof.localId() << " value=" << elimination_value;
339 _setMatrixValue(dof, dof, 1.0);
340 }
341 }
342}

346void AlephDoFLinearSystemImpl::
347_applyRHSTransformation()
348{
349 const bool do_print_filling = m_do_print_filling;
350
351 // Apply Row+Column elimination
352 // Phase 1:
353 // - subtract values of the RHS vector if Row+Column elimination
354 _applyRowColumnEliminationToRHS(do_print_filling);
355
356 IItemFamily* dof_family = dofFamily();
357
358 auto& dof_elimination_info = getEliminationInfo();
359 auto& dof_elimination_value = getEliminationValue();
360 auto& rhs_variable = rhsVariable();
361
362 // Apply Row or Row+Column elimination on RHS
363 ENUMERATE_ (DoF, idof, dof_family->allItems()) {
364 DoF dof = *idof;
365 if (!dof.isOwn())
366 continue;
367 Byte elimination_info = dof_elimination_info[dof];
368 if (elimination_info == ELIMINATE_ROW || elimination_info == ELIMINATE_ROW_COLUMN) {
369 Real elimination_value = dof_elimination_value[dof];
370 rhs_variable[dof] = elimination_value;
371 if (do_print_filling)
372 info() << "EliminateRHS info=" << static_cast<int>(elimination_info) << " row="
373 << std::setw(4) << dof.localId() << " value=" << elimination_value;
374 }
375 }

396void AlephDoFLinearSystemImpl::
397_fillRHSVector()
398{
399 ARCANE_CHECK_POINTER(m_aleph_rhs_vector);
400
401 // For the LinearSystem class we need an array
402 // with only the values for the ownNodes().
403 // The values of 'rhs_values' should not be updated after
404 // this call.
405 UniqueArray<Real> rhs_values_for_linear_system;
406 VariableDoFReal& rhs_values(rhsVariable());
407 IItemFamily* dof_family = dofFamily();
408 ENUMERATE_ (DoF, idof, dof_family->allItems().own()) {
409 Real v = rhs_values[idof];
410 if (m_do_print_filling)
411 info() << "SET VECTOR VALUE (" << std::setw(4) << idof.itemLocalId() << ") = " << v;
412 rhs_values_for_linear_system.add(rhs_values[idof]);
413 }
414
415 m_aleph_rhs_vector->setLocalComponents(rhs_values_for_linear_system.view());
416 m_aleph_rhs_vector->assemble();
417}

485void AlephDoFLinearSystemImpl::
486_computeMatrixInfo()
487{
488 int solver_backend = static_cast<int>(m_solver_backend);
489 info() << "[AlephFem] COMPUTE_MATRIX_INFO solver_backend=" << solver_backend;
490 IParallelMng* pm = m_sub_domain->parallelMng();
491 // Aleph solver:
492 // Hypre = 2
493 // Trilinos = 3
494 // Cuda = 4 (not available)
495 // Petsc = 5
496 // We need to compile Arcane with the needed library and link
497 // the code with the associated aleph library (see CMakeLists.txt)
498 // TODO: Linear algebra backend should be accessed from arc file.
499 if (!m_aleph_kernel) {
500 info() << "Creating Aleph Kernel";
501 // We can use less than the number of MPI ranks
502 // but for the moment we use all the available cores.
503 Int32 nb_core = pm->commSize();
504 m_aleph_kernel = new AlephKernel(m_sub_domain, solver_backend, nb_core);
505 }
506 else {
507 //
508 m_need_destroy_matrix_and_vector = false;
509 }
510 IItemFamily* dof_family = dofFamily();
511 VariableDoFReal& solution_variable(solutionVariable());
512 DoFGroup own_dofs = dof_family->allItems().own();
513 m_dof_matrix_indexes.fill(-1);
514 AlephIndexing* indexing = m_aleph_kernel->indexing();
515 ENUMERATE_ (DoF, idof, own_dofs) {
516 DoF dof = *idof;
517 Integer row = indexing->get(solution_variable, dof);
518 m_dof_matrix_indexes[dof] = row;
519 }
520
521 // Do not print information about setting matrix if matrix is too big
522 if (own_dofs.size() > 200)
523 m_do_print_filling = false;
524}

556
557void AlephDoFLinearSystemImpl::
558solve()
559{
560 // The matrix always need to be created if this is not explicitly done
561 if (!m_aleph_matrix)
562 applyMatrixTransformation();
563
564 _createRHSAndSolutionVector();
565 _fillRHSVector();
566 _fillSolutionVector();
567
568 info() << "Calling AlephDoFLinearSystemImpl::solve()";
569 UniqueArray<Real> aleph_result;
570
571 IItemFamily* dof_family = dofFamily();
572 DoFGroup own_dofs = dof_family->allItems().own();
573 const Int32 nb_dof = own_dofs.size();
574
575 Int32 nb_iteration = 0;
576 Real residual_norm = 0.0;
577 info() << "[AlephFem] BEGIN SOLVING WITH ALEPH solver_backend=" << static_cast<int>(m_solver_backend);
578
579 // Post the solver. The call is asynchronous, and we wait for the result
580 // when calling syncSolver().
581 // The values nb_iteration and residual_norm are not used in this case.
582 // We get them during the call to syncSolver()
583 m_aleph_matrix->solve(m_aleph_solution_vector, m_aleph_rhs_vector,
584 nb_iteration, &residual_norm,
585 m_aleph_params, true);
586
587 // Reset matrix and vectors because there can no longer be used
588 // They will be re-created when needed if we call solve() again.
589 // NOTE: it is the aleph library we do not need to call delete() because
590 m_aleph_rhs_vector = nullptr;
591 m_aleph_solution_vector = nullptr;
592 m_aleph_matrix = nullptr;
593
594 info() << "[AlephFem] END SOLVING WITH ALEPH r=" << residual_norm
595 << " nb_iter=" << nb_iteration;
596
597 // Wait for the solver to finish and get solution vector
598 auto* solution_vector = m_aleph_kernel->syncSolver(0, nb_iteration, &residual_norm);
599
600 solution_vector->getLocalComponents(aleph_result);
601
602 const bool do_verbose = (nb_dof < 200);
603 Int32 index = 0;
604
605 VariableDoFReal& solution_variable(this->solutionVariable());
606 ENUMERATE_ (DoF, idof, dofFamily()->allItems().own()) {
607 DoF dof = *idof;
608
609 solution_variable[dof] = aleph_result[m_aleph_kernel->indexing()->get(solution_variable, dof)];
610 if (do_verbose)
611 info() << "Node uid=" << dof.uniqueId() << " V=" << aleph_result[index] << " T=" << solution_variable[dof];
612 ++index;
613 }
614}
