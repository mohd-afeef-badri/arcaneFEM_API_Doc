15/**
16 * @brief Computes the element matrix for a quadrilateral element (QUAD4, ℙ1 FE).
17 *
18 * This function calculates the integral of:
19 * 𝑎(𝑢,𝑣) = ∫∫ λ(∂𝑢/∂𝑥 ∂𝑣/∂𝑥 + ∂𝑢/∂𝑦 ∂𝑣/∂𝑦)dΩ + ∫∫ (𝑢𝑣/δ𝑡)dΩ
20 *
21 * Steps involved:
22 * 1. Define Gauss points (2x2) and weights.
23 * 2. Loop over Gauss points to compute the gradients in physical space
24 * and the determinant of the Jacobian, via computeGradientsAndJacobianQuad4.
25 * 3. Compute the integration weight.
26 * 4. Assemble the element matrix using the computed gradients.
27 *
28 * @param cell The cell for which the element matrix is computed.
29 * @return The computed element matrix.
30 */
31/*---------------------------------------------------------------------------*/
32
33RealMatrix<4, 4> FemModuleHeat::_computeElementMatrixQuad4(Cell cell)
34{
35 // Gauss points and weights for 2x2 quadrature
36 constexpr Real gp[2] = { -M_SQRT1_3, M_SQRT1_3 }; // [-1/sqrt(3) , 1/sqrt(3)]
37 constexpr Real w = 1.0;
38 const Real lambda = m_cell_lambda[cell];
39
40 // Initialize the element matrix
41 RealMatrix<4, 4> ae;
42 ae.fill(0.0);
43
44 // Loop over Gauss points
45 for (Int8 ixi = 0; ixi < 2; ++ixi) {
46 for (Int8 ieta = 0; ieta < 2; ++ieta) {
47
48 // Get the coordinates of the Gauss point in natural coordinates (ξ,η,ζ)
49 const Real xi = gp[ixi];
50 const Real eta = gp[ieta];
51
52 // Get shape function gradients w.r.t (𝑥,𝑦) and determinant of Jacobian
53 const auto gp_info = ArcaneFemFunctions::FeOperation2D::computeGradientsAndJacobianQuad4(cell, m_node_coord, xi, eta);
54 const RealVector<4>& dxU = gp_info.dN_dx;
55 const RealVector<4>& dyU = gp_info.dN_dy;
56 const Real detJ = gp_info.det_j;
57
58 // Shape functions at the Gauss point
59 RealVector<4> N = ArcaneFemFunctions::FeOperation2D::computeShapeFunctionsQuad4(xi, eta);
60
61 // Integration weight
62 const Real integration_weight = detJ * w * w;
63
64 // stiffness matrix assembly
65 ae += (dxU ^ dxU) * integration_weight * lambda + (dyU ^ dyU) * integration_weight * lambda +
66 (N ^ N) * integration_weight / dt; // mass matrix contribution
67 }
68 }
69 return ae;
70}

73/**
74 * @brief Computes the element matrix for a hexahedral element (HEXA8, ℙ1 FE).
75 *
76 * This function calculates the integral of:
77 * 𝑎(𝑢,𝑣) = ∫∫∫ λ(∂𝑢/∂𝑥 ∂𝑣/∂𝑥 + ∂𝑢/∂𝑦 ∂𝑣/∂𝑦 + ∂𝑢/∂𝑧 ∂𝑣/∂𝑧)dΩ + ∫∫∫ (𝑢𝑣/δ𝑡)dΩ
78 *
79 * Steps involved:
80 * 1. Define Gauss points (2x2x2) and weights.
81 * 2. Loop over Gauss points to compute the gradients in physical space (𝑥,𝑦,𝑧)
82 * and the determinant of the Jacobian via computeGradientsAndJacobianHexa8.
83 * 3. Compute the integration weight.
84 * 4. Assemble the element matrix using the computed gradients.
85 *
86 * @param cell The cell for which the element matrix is computed.
87 * @return The computed element matrix.
88 */
89/*---------------------------------------------------------------------------*/
90
91RealMatrix<8, 8> FemModuleHeat::_computeElementMatrixHexa8(Cell cell)
92{
93 // 2x2x2 Gauss points and weights for [-1,1]^3
94 constexpr Real gp[2] = { -M_SQRT1_3, M_SQRT1_3 }; // -1/sqrt(3), 1/sqrt(3)
95 constexpr Real w = 1.0;
96 const Real lambda = m_cell_lambda[cell];
97
98 // Initialize the element matrix
99 RealMatrix<8, 8> ae;
100 ae.fill(0.0);
101
102 // Loop over Gauss points
103 for (Int8 ixi = 0; ixi < 2; ++ixi) {
104 for (Int8 ieta = 0; ieta < 2; ++ieta) {
105 for (Int8 izeta = 0; izeta < 2; ++izeta) {
106
107 // Get the coordinates of Gauss points in natural coordinates (ξ,η,ζ)
108 const Real xi = gp[ixi];
109 const Real eta = gp[ieta];
110 const Real zeta = gp[izeta];
111
112 // Get shape function gradients w.r.t (𝑥,𝑦,𝑧) and determinant of Jacobian
113 const auto gp_info = ArcaneFemFunctions::FeOperation3D::computeGradientsAndJacobianHexa8(cell, m_node_coord, xi, eta, zeta);
114 const RealVector<8>& dxU = gp_info.dN_dx;
115 const RealVector<8>& dyU = gp_info.dN_dy;
116 const RealVector<8>& dzU = gp_info.dN_dz;
117 const Real detJ = gp_info.det_j;
118
119 // Shape functions at the Gauss point
120 RealVector<8> N = ArcaneFemFunctions::FeOperation3D::computeShapeFunctionsHexa8(xi, eta, zeta);
121
122 // Integration weight
123 const Real integration_weight = detJ * w * w;
124
125 // Assemble element matrix (variational form)
126 ae += (dxU ^ dxU) * integration_weight * lambda + (dyU ^ dyU) * integration_weight * lambda + (dzU ^ dzU) * integration_weight * lambda
127 + (N ^ N) * integration_weight / dt; // mass matrix contribution
128 }
129 }
130 }
131 return ae;
132}
