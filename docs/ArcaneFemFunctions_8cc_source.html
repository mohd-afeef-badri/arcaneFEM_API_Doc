63/**
64 * @brief Provides at once, all Gauss data of a given input finite element
65 * (vector containing weights, ref. coordinates, nodal shape values & derivatives)
66 * This method is generic (coming from PASSMO)
67 */
68/*---------------------------------------------------------------------------*/
69RealUniqueArray ArcaneFemFunctions::CellFEMDispatcher::
70getGaussData(ItemWithNodes item, Integer nint, Integer ngauss){
71
72 auto nnod = item.nbNode();
73 auto cell_type = item.type();
74 ngauss = ArcaneFemFunctions::FemGaussQuadrature::getNbGaussPointsfromOrder(cell_type,nint);
75
76 // Vector of double containing:
77 // ngauss points * [weight, gauss ref coord [Real3], nnod * (shapefunc values, 3*shapefunc deriv
78 // in ref. coord system)]
79 Integer nsize = ngauss * 4 * (1 + nnod);
80 RealUniqueArray vec(nsize);
81
82 Integer index{ 0 };
83 for (Integer ig = 0; ig < ngauss; ++ig) {
84 auto wt = ArcaneFemFunctions::FemGaussQuadrature::getGaussWeight(item, nint, ig);
85 auto pos = ArcaneFemFunctions::FemGaussQuadrature::getGaussRefPosition(item, nint, ig);
86 vec[index++] = wt;
87 vec[index++] = pos.x;
88 vec[index++] = pos.y;
89 vec[index++] = pos.z;
90
91 for (Int32 inod = 0; inod < nnod; ++inod) {
92 auto Phi_i = getShapeFuncVal(cell_type, inod, pos);
93 vec[index++] = Phi_i;
94 auto dPhi = getShapeFuncDeriv(cell_type, inod, pos);
95 vec[index++] = dPhi.x;
96 vec[index++] = dPhi.y;
97 vec[index++] = dPhi.z;
98 }
99 }
100 return vec;
101}
