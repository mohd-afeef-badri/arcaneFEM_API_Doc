45/*---------------------------------------------------------------------------*/
46/**
47 * @brief Performs the main computation for the FemModuleElasticity.
48 *
49 * This method:
50 * 1. Stops the time loop after 1 iteration since the equation is steady state.
51 * 2. Resets, configures, and initializes the linear system.
52 * 3. Sets Petsc flags if user has provided them.
53 * 4. Executes the stationary solve.
54 */
55/*---------------------------------------------------------------------------*/
56
57void FemModuleElasticity::
58compute()
59{
60 info() << "[ArcaneFem-Info] Started module compute()";
61 Real elapsedTime = platform::getRealTime();
62
63 // Stop code after computations
64 if (m_global_iteration() > 0)
65 subDomain()->timeLoopMng()->stopComputeLoop(true);
66
67 m_linear_system.reset();
68 m_linear_system.setLinearSystemFactory(options()->linearSystem());
69 m_linear_system.initialize(subDomain(), acceleratorMng()->defaultRunner(), m_dofs_on_nodes.dofFamily(), "Solver");
70 m_linear_system.clearValues();
71
72 if (m_petsc_flags != NULL){
73 CommandLineArguments args = ArcaneFemFunctions::GeneralFunctions::getPetscFlagsFromCommandline(m_petsc_flags);
74 m_linear_system.setSolverCommandLineArguments(args);
75 }
76
77 if (m_matrix_format == "BSR" || m_matrix_format == "AF-BSR")
78 _initBsr();
79
80 Int64 nb_node = mesh()->ownNodes().size();
81 Int64 total_nb_node = mesh()->parallelMng()->reduce(Parallel::ReduceSum, nb_node);
82
83 Int64 nb_face = mesh()->outerFaces().size();
84 Int64 total_nb_boundary_elt = mesh()->parallelMng()->reduce(Parallel::ReduceSum, nb_face);
85
86 Int64 nb_cell = mesh()->ownCells().size();
87 Int64 total_nb_elt = mesh()->parallelMng()->reduce(Parallel::ReduceSum, nb_cell);
88
89 info() << "[ArcaneFem-Info] mesh dimension " << defaultMesh()->dimension();
90 info() << "[ArcaneFem-Info] mesh boundary elements " << total_nb_boundary_elt;
91 info() << "[ArcaneFem-Info] mesh cells " << total_nb_elt;
92 info() << "[ArcaneFem-Info] mesh nodes " << total_nb_node;
93
94 _doStationarySolve();
95
96 elapsedTime = platform::getRealTime() - elapsedTime;
97 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(),"compute", elapsedTime);
98}

125/*---------------------------------------------------------------------------*/
126/**
127 * @brief Performs a stationary solve for the FEM system.
128 *
129 * This method follows a sequence of steps to solve FEM system:
130 *
131 * 1. _getMaterialParameters() Retrieves material parameters via
132 * 2. _assembleBilinearOperator() Assembles the FEM matrix 𝐀
133 * 3. _assembleLinearOperator() Assembles the FEM RHS vector 𝐛
134 * 4. _solve() Solves for solution vector 𝐮 = 𝐀⁻¹𝐛
135 * 5. _updateVariables() Updates FEM variables 𝐮 = 𝐱
136 * 6. _validateResults() Regression test
137 */
138/*---------------------------------------------------------------------------*/
139
140void FemModuleElasticity::
141_doStationarySolve()
142{
143 if(m_assemble_linear_system){
144 _assembleBilinearOperator();
145 _assembleLinearOperator();
146 }
147 if(m_solve_linear_system){
148 _solve();
149 _updateVariables();
150 }
151 if(m_cross_validation){
152 _validateResults();
153 }
154}

161
162void FemModuleElasticity::
163_getMaterialParameters()
164{
165 info() << "[ArcaneFem-Info] Started module _getMaterialParameters()";
166 Real elapsedTime = platform::getRealTime();
167
168 E = options()->E(); // Youngs modulus
169 nu = options()->nu(); // Poission ratio ν
170
171 mu = (E / (2 * (1 + nu))); // lame parameter μ
172 lambda = E * nu / ((1 + nu) * (1 - 2 * nu)); // lame parameter λ
173
174 m_dof_per_node = defaultMesh()->dimension();
175 m_matrix_format = options()->matrixFormat();
176 m_assemble_linear_system = options()->assembleLinearSystem();
177 m_solve_linear_system = options()->solveLinearSystem();
178 m_cross_validation = options()->hasSolutionComparisonFile();
179 m_petsc_flags = options()->petscFlags();
180 m_hex_quad_mesh = options()->hexQuadMesh();
181
182 elapsedTime = platform::getRealTime() - elapsedTime;
183 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(),"get-material-params", elapsedTime);
184}

191/*---------------------------------------------------------------------------*/
192/**
193 * @brief Assemble the FEM linear operator.
194 *
195 * This method follows a sequence of steps to assemble RHS of FEM linear system:
196 *
197 * 1. assembles the bodyforce contribution (source term) ∫∫∫ (𝐟.𝐯) on Ω
198 * 2. assembles the traction contribution (Neumann term) ∫∫ (𝐭.𝐯) on ∂Ω
199 * 3. apply Dirichlet contributions to LHS and RHS
200 */
201/*---------------------------------------------------------------------------*/
202
203void FemModuleElasticity::
204_assembleLinearOperator()
205{
206 info() << "[ArcaneFem-Info] Started module _assembleLinearOperator()";
207 Real elapsedTime = platform::getRealTime();
208
209 VariableDoFReal& rhs_values(m_linear_system.rhsVariable()); // Temporary variable to keep values for the RHS
210 rhs_values.fill(0.0);
211
212 auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
213
214 _applyBodyForce(rhs_values, node_dof);
215 _applyTraction(rhs_values, node_dof);
216 _applyDirichlet(rhs_values, node_dof);
217
218 elapsedTime = platform::getRealTime() - elapsedTime;
219 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(),"rhs-vector-assembly", elapsedTime);
220}

223/**
224 * @brief Calls the right function for LHS assembly given as mesh type.
225 */
226/*---------------------------------------------------------------------------*/
227
228void FemModuleElasticity::
229_assembleBilinearOperator()
230{
231 info() << "[ArcaneFem-Info] Started module _assembleBilinearOperator()";
232 Real elapsedTime = platform::getRealTime();
233
234 if (m_matrix_format == "BSR") {
235 UnstructuredMeshConnectivityView m_connectivity_view(mesh());
236 auto cn_cv = m_connectivity_view.cellNode();
237 auto command = makeCommand(acceleratorMng()->defaultQueue());
238 auto in_node_coord = Accelerator::viewIn(command, m_node_coord);
239 auto lambda_copy = lambda;
240 auto mu_copy = mu;
241
242 m_bsr_format.computeSparsity();
243 if (mesh()->dimension() == 2)
244 m_bsr_format.assembleBilinearAtomic([=] ARCCORE_HOST_DEVICE(CellLocalId cell_lid) { return computeElementMatrixTria3Gpu(cell_lid, cn_cv, in_node_coord, lambda_copy, mu_copy); });
245 if (mesh()->dimension() == 3)
246 m_bsr_format.assembleBilinearAtomic([=] ARCCORE_HOST_DEVICE(CellLocalId cell_lid) { return computeElementMatrixTetra4Gpu(cell_lid, cn_cv, in_node_coord, lambda_copy, mu_copy); });
247 m_bsr_format.toLinearSystem(m_linear_system);
248 }
249 else if (m_matrix_format == "AF-BSR") {
250 UnstructuredMeshConnectivityView m_connectivity_view(mesh());
251 auto cn_cv = m_connectivity_view.cellNode();
252 auto command = makeCommand(acceleratorMng()->defaultQueue());
253 auto in_node_coord = Accelerator::viewIn(command, m_node_coord);
254 auto lambda_copy = lambda;
255 auto mu_copy = mu;
256
257 m_bsr_format.computeSparsity();
258 if (mesh()->dimension() == 2)
259 m_bsr_format.assembleBilinearAtomicFree([=] ARCCORE_HOST_DEVICE(CellLocalId cell_lid, Int32 node_lid) { return computeElementVectorTria3Gpu(cell_lid, cn_cv, in_node_coord, lambda_copy, mu_copy, node_lid); });
260 if (mesh()->dimension() == 3)
261 m_bsr_format.assembleBilinearAtomicFree([=] ARCCORE_HOST_DEVICE(CellLocalId cell_lid, Int32 node_lid) { return computeElementVectorTetra4Gpu(cell_lid, cn_cv, in_node_coord, lambda_copy, mu_copy, node_lid); });
262 m_bsr_format.toLinearSystem(m_linear_system);
263 }
264 else if (m_matrix_format == "DOK") {
265 if (mesh()->dimension() == 2) {
266 if (m_hex_quad_mesh) {
267 _assembleBilinearOperatorCpu<8>([this](const Cell& cell) { return _computeElementMatrixQuad4(cell); });
268 }
269 else {
270 _assembleBilinearOperatorCpu<6>([this](const Cell& cell) { return _computeElementMatrixTria3(cell); });
271 }
272 }
273 if (mesh()->dimension() == 3) {
274 if (m_hex_quad_mesh) {
275 _assembleBilinearOperatorCpu<24>([this](const Cell& cell) { return _computeElementMatrixHexa8(cell); });
276 }
277 else {
278 _assembleBilinearOperatorCpu<12>([this](const Cell& cell) { return _computeElementMatrixTetra4(cell); });
279 }
280 }
281 }
282 else {
283 ARCANE_FATAL("Unsupported matrix type, only DOK| BSR|AF-BSR is supported.");
284 }
285
286 elapsedTime = platform::getRealTime() - elapsedTime;
287 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(),"lhs-matrix-assembly", elapsedTime);
288}

290/*---------------------------------------------------------------------------*/
291/**
292 * @brief Assembles the FEM bilinear operator on CPU.
293 *
294 * This method assembles the FEM stiffness matrix by iterating over each cell,
295 * computing the element stiffness matrix using the provided function, and
296 * populating the global stiffness matrix accordingly.
297 *
298 * @tparam N Total DOF size (nodes_per_element × dimensions).
299 * @param compute_element_matrix function computing cell's element stiffness matrix.
300 */
301/*---------------------------------------------------------------------------*/
302
303template <int N>
304void FemModuleElasticity::
305_assembleBilinearOperatorCpu(const std::function<RealMatrix<N, N>(const Cell&)>& compute_element_matrix)
306{
307 const Int32 dim = mesh()->dimension();
308 auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
309
310 ENUMERATE_ (Cell, icell, allCells()) {
311 Cell cell = *icell;
312 auto K_e = compute_element_matrix(cell);
313
314 Int32 n1_index = 0;
315 for (Node node1 : cell.nodes()) {
316 if (node1.isOwn()) {
317 Int32 n2_index = 0;
318 for (Node node2 : cell.nodes()) {
319 for (Int32 i = 0; i < dim; ++i) {
320 DoFLocalId dof1 = node_dof.dofId(node1, i);
321 for (Int32 j = 0; j < dim; ++j) {
322 DoFLocalId dof2 = node_dof.dofId(node2, j);
323 Real value = K_e(dim * n1_index + i, dim * n2_index + j);
324 m_linear_system.matrixAddValue(dof1, dof2, value);
325 }
326 }
327 ++n2_index;
328 }
329 }
330 ++n1_index;
331 }
332 }
333}

384/*---------------------------------------------------------------------------*/
385/**
386 * @brief Update the FEM variables.
387 *
388 * This method performs the following actions:
389 * 1. Fetches values of solution from solved linear system to FEM variables,
390 * i.e., it copies RHS DOF to u.
391 * 2. Performs synchronize of FEM variables across subdomains.
392 */
393/*---------------------------------------------------------------------------*/
394
395void FemModuleElasticity::
396_updateVariables()
397{
398 info() << "[ArcaneFem-Info] Started module _updateVariables()";
399 Real elapsedTime = platform::getRealTime();
400
401 {
402 VariableDoFReal& dof_u(m_linear_system.solutionVariable());
403 auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
404 if (mesh()->dimension() == 3)
405 ENUMERATE_ (Node, inode, ownNodes()) {
406 Node node = *inode;
407 Real u1_val = dof_u[node_dof.dofId(node, 0)];
408 Real u2_val = dof_u[node_dof.dofId(node, 1)];
409 Real u3_val = dof_u[node_dof.dofId(node, 2)];
410 m_U[node] = Real3(u1_val, u2_val, u3_val);
411 }
412 else
413 ENUMERATE_ (Node, inode, ownNodes()) {
414 Node node = *inode;
415 Real u1_val = dof_u[node_dof.dofId(node, 0)];
416 Real u2_val = dof_u[node_dof.dofId(node, 1)];
417 m_U[node] = Real3(u1_val, u2_val, 0.);
418 }
419 }
420
421 m_U.synchronize();
422
423 elapsedTime = platform::getRealTime() - elapsedTime;
424 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(),"update-variables", elapsedTime);
425}
