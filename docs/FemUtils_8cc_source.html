36void _convertNumArrayToCSRMatrix(Matrix& out_matrix, MDSpan<const Real, MDDim2> in_matrix)
37{
38 const Int32 matrix_size = in_matrix.extent0();
39 if (matrix_size != in_matrix.extent1())
40 ARCANE_FATAL("Not a square matrix ({0},{1})", matrix_size, in_matrix.extent1());
41 UniqueArray<Int32> nb_non_zero_in_row(matrix_size);
42
43 Int32 nb_non_zero = 0;
44 for (Int32 i = 0; i < matrix_size; ++i) {
45 Int32 row_non_zero = 0;
46 for (Int32 j = 0; j < matrix_size; ++j)
47 if (in_matrix(i, j) != 0)
48 ++row_non_zero;
49 nb_non_zero_in_row[i] = row_non_zero;
50 nb_non_zero += row_non_zero;
51 }
52
53 UniqueArray<Int32> columns(nb_non_zero);
54 UniqueArray<Real> values(nb_non_zero);
55
56 std::ostringstream ostr;
57 ostr.flags(ios::scientific);
58 ostr.precision(17);
59 {
60 Int32 fill_index = 0;
61 for (Int32 i = 0; i < matrix_size; ++i) {
62 for (Int32 j = 0; j < matrix_size; ++j) {
63 Real v = in_matrix(i, j);
64 if (matrix_size<200 && v!=0.0)
65 ostr << "MAT[" << i << "][" << j << "] = " << v << endl;
66 if (v != 0) {
67 columns[fill_index] = j;
68 values[fill_index] = v;
69 ++fill_index;
70 }
71 }
72 }
73 }
74 std::cout << ostr.str();
75 out_matrix.setRowsSize(nb_non_zero_in_row);
76 out_matrix.setValues(columns, values);
77}

85 inline bool
86 _isDifferent(Real ref_v, Real v, Real epsilon, Real min_value)
87 {
88 if (math::abs(ref_v) < min_value && math::abs(v) < min_value)
89 return false;
90 return !TypeEqualT<Real>::isNearlyEqualWithEpsilon(ref_v, v, epsilon);
91 }
92 inline bool
93 _isDifferent(Real2 ref_v, Real2 v, Real epsilon, Real min_value)
94 {
95 return _isDifferent(ref_v.x, v.x, epsilon, min_value) || _isDifferent(ref_v.y, v.y, epsilon, min_value);
96 }
97 inline bool
98 _isDifferent(Real3 ref_v, Real3 v, Real epsilon, Real min_value)
99 {
100 return _isDifferent(ref_v.x, v.x, epsilon, min_value) ||
101 _isDifferent(ref_v.y, v.y, epsilon, min_value) ||
102 _isDifferent(ref_v.z, v.z, epsilon, min_value);
103 }

105 template <typename VariableType> inline void
106 _checkNodeResultFile(ITraceMng* tm, const String& filename,
107 const VariableType& node_values,
108 double epsilon, double min_value)
109 {
110 ARCANE_CHECK_POINTER(tm);
111 using DataType = typename VariableType::DataType;
112 tm->info() << "CheckNodeResultFile filename=" << filename << " epsilon=" << epsilon
113 << " min_value=" << min_value;
114 if (filename.empty())
115 ARCANE_FATAL("Invalid empty filename");
116 IItemFamily* node_family = node_values.variable()->itemFamily();
117 if (!node_family)
118 ARCANE_FATAL("Variable '{0}' is not allocated", node_values.name());
119
120 std::map<Int64, DataType> item_reference_values;
121 {
122 std::ifstream sbuf(filename.localstr());
123 DataType read_value{};
124 Int64 read_uid = 0;
125 if (!sbuf.eof())
126 sbuf >> ws;
127 while (!sbuf.eof()) {
128 sbuf >> read_uid >> ws >> read_value;
129 //tm->info() << " R uid=" << read_uid << " v=" << read_value;
130 if (sbuf.fail() || sbuf.bad())
131 ARCANE_FATAL("Error during parsing of file '{0}'", filename);
132 item_reference_values.insert(std::make_pair(read_uid, read_value));
133 sbuf >> ws;
134 }
135 }
136
137 tm->info() << "NB_Values=" << item_reference_values.size();
138
139 // Get Max UID
140 Int64 max_uid = 0;
141 ENUMERATE_ (Node, inode, node_family->allItems()) {
142 Node node = *inode;
143 if (node.uniqueId() > max_uid)
144 max_uid = node.uniqueId();
145 }
146
147 std::map<Int64, DataType> item_current_values;
148
149 ENUMERATE_ (Node, inode, node_family->allItems()) {
150 Node node = *inode;
151 item_current_values[node.uniqueId()] = node_values[node];
152 }
153
154 Int64 nb_error = 0;
155 for (const auto& x : item_current_values) {
156 Int64 uid = x.first;
157 DataType v = x.second;
158 auto x_ref = item_reference_values.find(uid);
159 if (x_ref != item_reference_values.end()) {
160 DataType ref_v = x_ref->second;
161 if (_isDifferent(ref_v, v, epsilon, min_value)) {
162 ++nb_error;
163 if (nb_error < 50)
164 tm->info() << String::format("ERROR: uid={0} ref={1} v={2} diff={3}",
165 uid, ref_v, v, ref_v - v);
166 }
167 }
168 }
169 if (nb_error > 0)
170 ARCANE_FATAL("Error checking values nb_error={0}", nb_error);
171 }

176/*!
177 * \brief Sample to read value from a file and create an associated CaseTable.
178 *
179 * The file should contains 3 values for each time step (so the number of
180 * values should be a multiple of 4).
181 */
182CaseTable* readFileAsCaseTable(IParallelMng* pm, const String& filename, const Int32& ndim)
183{
184 ITraceMng* tm = pm->traceMng();
185 UniqueArray<Byte> bytes;
186 bool is_bad = pm->ioMng()->collectiveRead(filename, bytes, false);
187 if (is_bad)
188 ARCANE_FATAL("Can not read file '{0}'");
189 String file_as_str(bytes);
190 tm->info() << "FILE=" << file_as_str;
191 UniqueArray<Real> file_values;
192 is_bad = builtInGetValue(file_values, file_as_str);
193 if (is_bad)
194 ARCANE_FATAL("Can not read file '{0}' as Array<Real>");
195 Int32 nb_value = file_values.size();
196
197 // For each line, the first value is the time step and the 3 following values
198 // are the values of the case table. So the total number should be a multiple of 4
199 tm->info() << "NB_VALUE=" << file_values.size();
200 Int32 nb_func_value = nb_value / (ndim+1);
201 tm->info() << "NB_FUNC_VALUE=" << nb_func_value;
202 if ((nb_func_value * (ndim+1)) != nb_value)
203 ARCANE_FATAL("Bad number of values: {0} should be a multiple of {1}", nb_value, ndim+1);
204
205 CaseFunctionBuildInfo cfbi(tm, "MyTestTable");
206 cfbi.m_param_type = ICaseFunction::ParamReal;
207 cfbi.m_value_type = ICaseFunction::ValueReal3;
208
209 // Create the associated CaseTable and fill it with the values from the file
210 auto table = new CaseTable(cfbi, CaseTable::CurveLinear);
211 for (Int32 i = 0; i < nb_func_value; ++i) {
212 Int32 index = i * (ndim+1);
213 Real func_time = file_values[index];
214 Real3 v;
215 for (int j = 0; j < ndim; ++j) {
216 v[j] = file_values[index + j + 1];
217 }
218 String func_param = String::fromNumber(func_time);
219 String func_value;
220 switch (ndim) {
221 default:
222 case 1: func_value = String::format("{0}", v[0]); break;
223 case 2: func_value = String::format("{0} {1}", v[0], v[1]); break;
224 case 3: func_value = String::format("{0} {1} {2}", v[0], v[1], v[2]); break;
225 }
226 table->appendElement(func_param, func_value);
227 }
228 return table;
229}

264Real real3x3Trace(const Real3x3& mat) {
265 return mat[0][0] + mat[1][1] + mat[2][2];
266}
267
268/*---------------------------------------------------------------------------*/
269/*---------------------------------------------------------------------------*/
270
271Real3 real3x3GetSupOutdiagonal(const Real3x3& mat) {
272 return {mat[0][1], mat[0][2], mat[1][2]};
273}
274
275/*---------------------------------------------------------------------------*/
276/*---------------------------------------------------------------------------*/
277
278Real3 real3x3GetLowOutdiagonal(const Real3x3& mat){
279 return {mat[1][0], mat[2][0], mat[2][1]};
280}
281
282/*---------------------------------------------------------------------------*/
283/*---------------------------------------------------------------------------*/
284Real3 diagonalReal3x3(const Real3x3& mat){
285 return { mat[0][0], mat[1][1], mat[2][2]} ;
286}
287
288/*---------------------------------------------------------------------------*/
289/*---------------------------------------------------------------------------*/
290Real3x3 outdiagonalReal3x3(const Real3x3& mat){
291 Real3 diag = diagonalReal3x3(mat);
292 Real3x3 mdiag;
293 mdiag.x.x = diag.x;
294 mdiag.y.y = diag.y;
295 mdiag.z.z = diag.z;
296 return (mat - mdiag);
297}

301
302bool real3x3IsSym(const Real3x3& mat)
303{
304 Real3 matsup = real3x3GetSupOutdiagonal(mat);
305 Real3 matlow = real3x3GetLowOutdiagonal(mat);
306 return (matsup == matlow);
307}
