23ARCCORE_HOST_DEVICE static UInt64 pack(Int32 n0, Int32 n1)
24{
25 Int32 min = n0 > n1 ? n1 : n0;
26 Int32 max = n0 > n1 ? n0 : n1;
27 return ((UInt64)min << 32) | (UInt64)max;
28}

33ARCCORE_HOST_DEVICE static void unpack(UInt64 packed_edge, Int32& n0, Int32& n1)
34{
35 n0 = (Int32)(packed_edge >> 32);
36 n1 = (Int32)(packed_edge & 0xFFFFFFFF);
37}

42void FemModuleTestlab::_computeSortedEdges(Int8 edges_per_element, Int64 nb_edge_total, SmallSpan<UInt64>& sorted_edges_ss)
43{
44 auto mem_ressource = m_queue.memoryRessource();
45 NumArray<UInt64, MDDim1> edges(mem_ressource);
46 edges.resize(nb_edge_total);
47
48 UnstructuredMeshConnectivityView m_connectivity_view(mesh());
49 auto cell_node_cv = m_connectivity_view.cellNode();
50
51 {
52 auto command = makeCommand(m_queue);
53 auto inout_edges = viewInOut(command, edges);
54
55 if (mesh()->dimension() == 2) {
56 command << RUNCOMMAND_ENUMERATE(CellLocalId, cell_lid, mesh()->allCells())
57 {
58 auto n0 = cell_node_cv.nodeId(cell_lid, 0);
59 auto n1 = cell_node_cv.nodeId(cell_lid, 1);
60 auto n2 = cell_node_cv.nodeId(cell_lid, 2);
61
62 auto start = cell_lid * edges_per_element;
63 inout_edges[start] = pack(n0, n1);
64 inout_edges[start + 1] = pack(n0, n2);
65 inout_edges[start + 2] = pack(n1, n2);
66 };
67 }
68 else {
69 command << RUNCOMMAND_ENUMERATE(CellLocalId, cell_lid, mesh()->allCells())
70 {
71 auto n0 = cell_node_cv.nodeId(cell_lid, 0);
72 auto n1 = cell_node_cv.nodeId(cell_lid, 1);
73 auto n2 = cell_node_cv.nodeId(cell_lid, 2);
74 auto n3 = cell_node_cv.nodeId(cell_lid, 3);
75
76 auto start = cell_lid * edges_per_element;
77 inout_edges[start] = pack(n0, n1);
78 inout_edges[start + 1] = pack(n0, n2);
79 inout_edges[start + 2] = pack(n0, n3);
80 inout_edges[start + 3] = pack(n1, n2);
81 inout_edges[start + 4] = pack(n1, n3);
82 inout_edges[start + 5] = pack(n2, n3);
83 };
84 }
85 }
86 m_queue.barrier();
87
88 auto sorter = Accelerator::GenericSorter(m_queue);
89 SmallSpan<const UInt64> edges_ss = edges.to1DSmallSpan();
90 sorter.apply(edges_ss, sorted_edges_ss);
91}

96void FemModuleTestlab::_computeNeighbors(Int8 edges_per_element, Int64 nb_edge_total, NumArray<Int32, MDDim1>& neighbors, SmallSpan<UInt64>& sorted_edges_ss)
97{
98 auto command = makeCommand(m_queue);
99 auto inout_neighbors = viewInOut(command, neighbors);
100
101 command << RUNCOMMAND_LOOP1(iter, nb_edge_total)
102 {
103 auto [thread_id] = iter();
104 auto cur_edge = sorted_edges_ss[thread_id];
105 if (thread_id == (nb_edge_total - 1) || cur_edge != sorted_edges_ss[thread_id + 1]) {
106 Int32 n0, n1 = 0;
107 unpack(cur_edge, n0, n1);
108 Accelerator::doAtomic<Accelerator::eAtomicOperation::Add>(inout_neighbors[n0], 1);
109 Accelerator::doAtomic<Accelerator::eAtomicOperation::Add>(inout_neighbors[n1], 1);
110 }
111 };
112}

117void FemModuleTestlab::_computeRowIndex(Int8 edges_per_element, Int64 nb_edge_total, SmallSpan<UInt64>& sorted_edges_ss) {
118 auto mem_ressource = m_queue.memoryRessource();
119 NumArray<Int32, MDDim1> neighbors(mem_ressource);
120 neighbors.resize(mesh()->nbNode());
121 neighbors.fill(1, &m_queue);
122 _computeNeighbors(edges_per_element, nb_edge_total, neighbors, sorted_edges_ss);
123
124 Accelerator::Scanner<Int32> scanner;
125 SmallSpan<Int32> neighbors_ss = neighbors.to1DSmallSpan();
126 scanner.exclusiveSum(&m_queue, neighbors_ss, m_csr_matrix.m_matrix_row.to1DSmallSpan());
127}

132ARCCORE_HOST_DEVICE static void registerEdgeInColumns(Int32 src, Int32 dst, Accelerator::NumArrayView<DataViewGetterSetter<Int32>, MDDim1, DefaultLayout> offsets, Accelerator::NumArrayView<DataViewGetter<Int32>, MDDim1, DefaultLayout> row_index, Accelerator::NumArrayView<DataViewGetterSetter<Int32>, MDDim1, DefaultLayout> columns)
133{
134 Int32 start = row_index[src];

136 columns[start + offset] = dst;

142void FemModuleTestlab::_computeColumns(Int8 edges_per_element, Int64 nb_edge_total, SmallSpan<uint64_t>& sorted_edges_ss) {
143 auto nb_node = mesh()->nbNode();
144
145 {
146 auto command = makeCommand(m_queue);
147 auto in_row_index = viewIn(command, m_csr_matrix.m_matrix_row);
148 auto inout_columns = viewInOut(command, m_csr_matrix.m_matrix_column);
149
150 command << RUNCOMMAND_LOOP1(iter, nb_node)
151 {
152 auto [n] = iter();
153 auto start = in_row_index[n];
154 inout_columns[start] = n;
155 };
156 }
157 m_queue.barrier();
158
159 auto mem_ressource = m_queue.memoryRessource();
160 NumArray<Int32, MDDim1> offsets(mem_ressource);
161 offsets.resize(nb_node);
162 offsets.fill(1, &m_queue);
163
164 {
165 auto command = makeCommand(m_queue);
166 auto inout_columns = viewInOut(command, m_csr_matrix.m_matrix_column);
167 auto in_row_index = viewIn(command, m_csr_matrix.m_matrix_row);
168 auto inout_offsets = viewInOut(command, offsets);
169
170 command << RUNCOMMAND_LOOP1(iter, nb_edge_total)
171 {
172 auto [thread_id] = iter();
173 auto cur_edge = sorted_edges_ss[thread_id];
174 if (thread_id == (nb_edge_total - 1) || cur_edge != sorted_edges_ss[thread_id + 1]) {
175 Int32 n0, n1 = 0;
176 unpack(cur_edge, n0, n1);
177 registerEdgeInColumns(n0, n1, inout_offsets, in_row_index, inout_columns);
178 registerEdgeInColumns(n1, n0, inout_offsets, in_row_index, inout_columns);
179 }
180 };
181 }

187void FemModuleTestlab::_computeSparsity()
188{
189
190 Int8 mesh_dim = mesh()->dimension();
191
192 Int32 nb_node = nbNode();
193 Int32 nb_non_zero = nb_node + 2 * (mesh_dim == 2 ? nbFace() : m_nb_edge);
194 m_csr_matrix.initialize(m_dof_family, nb_non_zero, nb_node, m_queue);
195
196 auto edges_per_element = mesh_dim == 2 ? 3 : 6;
197 auto nb_edge_total = mesh()->nbCell() * edges_per_element;
198
199 auto mem_ressource = m_queue.memoryRessource();
200 NumArray<UInt64, MDDim1> sorted_edges(mem_ressource);
201 sorted_edges.resize(nb_edge_total);
202 auto sorted_edges_ss = sorted_edges.to1DSmallSpan();
203
204 _computeSortedEdges(edges_per_element, nb_edge_total, sorted_edges_ss);
205 _computeRowIndex(edges_per_element, nb_edge_total, sorted_edges_ss);
206 _computeColumns(edges_per_element, nb_edge_total, sorted_edges_ss);
207}

210/**
211* @brief Assembles the bilinear operator matrix for the FEM linear system with
212 * the CSR sparse matrix format for TRIA3 elements.
213 *
214 * The method performs the following steps:
215 * 1. Builds the CSR matrix using _buildMatrixCsrGPU.
216 * 2. For each cell, computes element matrix using _computeElementMatrixTRIA3GPU.
217 * 3. Assembles global matrix by adding contributions from each cell's element
218 * matrix to the corresponding entries in the global matrix.
219 */
220/*---------------------------------------------------------------------------*/
221
222void FemModuleTestlab::
223_assembleCsrGPUBilinearOperatorTRIA3()
224{
225
226 Timer::Action timer_bili(m_time_stats, "AssembleBilinearOperator_Csr_Gpu");
227
228 {
229 Timer::Action timer_build(m_time_stats, "BuildMatrix");
230 // Build the csr matrix
231 _computeSparsity();
232 }
233
234 RunQueue* queue = acceleratorMng()->defaultQueue();
235 // Boucle sur les mailles déportée sur accélérateur
236 auto command = makeCommand(queue);
237
238 auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
239 auto in_row_csr = ax::viewIn(command, m_csr_matrix.m_matrix_row);
240 Int32 row_csr_size = m_csr_matrix.m_matrix_row.dim1Size();
241 auto in_col_csr = ax::viewIn(command, m_csr_matrix.m_matrix_column);
242 Int32 col_csr_size = m_csr_matrix.m_matrix_column.dim1Size();
243 auto in_out_val_csr = ax::viewInOut(command, m_csr_matrix.m_matrix_value);
244 UnstructuredMeshConnectivityView m_connectivity_view;
245 auto in_node_coord = ax::viewIn(command, m_node_coord);
246 m_connectivity_view.setMesh(this->mesh());
247 auto cnc = m_connectivity_view.cellNode();
248 Arcane::ItemGenericInfoListView nodes_infos(this->mesh()->nodeFamily());
249
250 Timer::Action timer_add_compute(m_time_stats, "AddAndCompute");
251
252 command << RUNCOMMAND_ENUMERATE(Cell, icell, allCells())
253 {
254
255 Real K_e[9] = { 0 };
256
257 _computeElementMatrixTRIA3GPU(icell, cnc, in_node_coord, K_e); // element stifness matrix
258 // # assemble elementary matrix into the global one
259 // # elementary terms are positionned into K according
260 // # to the rank of associated node in the mesh.nodes list
261 // for node1 in elem.nodes:
262 // inode1=elem.nodes.index(node1) # get position of node1 in nodes list
263 // for node2 in elem.nodes:
264 // inode2=elem.nodes.index(node2)
265 // K[node1.rank,node2.rank]=K[node1.rank,node2.rank]+K_e[inode1,inode2]
266 Int32 n1_index = 0;
267 for (NodeLocalId node1 : cnc.nodes(icell)) {
268 Int32 n2_index = 0;
269 for (NodeLocalId node2 : cnc.nodes(icell)) {
270 // K[node1.rank,node2.rank]=K[node1.rank,node2.rank]+K_e[inode1,inode2]
271 double v = K_e[n1_index * 3 + n2_index];
272 // m_k_matrix(node1.localId(), node2.localId()) += v;
273 if (nodes_infos.isOwn(node1)) {
274
275 Int32 row = node_dof.dofId(node1, 0).localId();
276 Int32 col = node_dof.dofId(node2, 0).localId();
277 Int32 begin = in_row_csr[row];
278 Int32 end = (row == row_csr_size - 1) ? col_csr_size : in_row_csr[row + 1];
279
280 while (begin < end) {
281 if (in_col_csr[begin] == col) {
282 // t is necessary to get the right type for the atomicAdd (but that means that we have more operations ?)
283 // The Macro is there to avoid compilation error if not in c++ 20
284 ax::doAtomic<ax::eAtomicOperation::Add>(in_out_val_csr(begin), v);
285 break;
286 }
287 begin++;
288 }
289 }
290 ++n2_index;
291 }
292 ++n1_index;
293 }
294 };
295}

298/**
299 * @brief Assembles the bilinear operator matrix for the FEM linear system with
300 * the CSR sparse matrix format for TETRA4 elements.
301 *
302 * The method performs the following steps:
303 * 1. Builds the CSR matrix using _buildMatrixCsrGPU.
304 * 2. For each cell, computes element matrix using _computeElementMatrixTETRA4GPU.
305 * 3. Assembles global matrix by adding contributions from each cell's element
306 * matrix to the corresponding entries in the global matrix.
307 */
308/*---------------------------------------------------------------------------*/
309
310void FemModuleTestlab::
311_assembleCsrGPUBilinearOperatorTETRA4()
312{
313 Timer::Action timer_bili(m_time_stats, "AssembleBilinearOperator_Csr_Gpu");
314
315 {
316 Timer::Action timer_build(m_time_stats, "BuildMatrix");
317 _computeSparsity();
318 }
319
320 {
321 auto command = makeCommand(m_queue);
322
323 Int32 row_csr_size = m_csr_matrix.m_matrix_row.extent0();
324 Int32 col_csr_size = m_csr_matrix.m_matrix_column.extent0();
325
326 auto in_row_csr = ax::viewIn(command, m_csr_matrix.m_matrix_row);
327 auto in_col_csr = ax::viewIn(command, m_csr_matrix.m_matrix_column);
328 auto inout_val_csr = ax::viewInOut(command, m_csr_matrix.m_matrix_value);
329
330 auto in_node_coord = ax::viewIn(command, m_node_coord);
331
332 UnstructuredMeshConnectivityView m_connectivity_view(mesh());
333 auto cell_node_connectivity_view = m_connectivity_view.cellNode();
334
335 ItemGenericInfoListView nodes_infos(mesh()->nodeFamily());
336
337 Timer::Action timer_add_compute(m_time_stats, "AddAndCompute");
338 ax::ProfileRegion ps_region(m_queue,"AddAndComputeBilinearTetra4",0x00FF7F);
339 command << RUNCOMMAND_ENUMERATE(Cell, icell, allCells())
340 {
341
342 Real K_e[16]{};
343 _computeElementMatrixTETRA4GPU(icell, cell_node_connectivity_view, in_node_coord, K_e);
344
345 Int32 node1_idx_in_cell = 0;
346 for (NodeLocalId node1_id : cell_node_connectivity_view.nodes(icell)) {
347
348 Int32 node2_idx_in_cell = 0;
349 for (NodeLocalId node2_id : cell_node_connectivity_view.nodes(icell)) {
350
351 if (nodes_infos.isOwn(node1_id)) {
352 double v = K_e[node1_idx_in_cell * 4 + node2_idx_in_cell];
353
354 Int32 row = node1_id.localId();
355 Int32 col = node2_id.localId();
356 Int32 begin = in_row_csr[row];
357
358 Int32 end = (row == row_csr_size - 1) ? col_csr_size : in_row_csr[row + 1];
359
360 while (begin < end) {
361 if (in_col_csr[begin] == col) {
362 ax::doAtomic<ax::eAtomicOperation::Add>(inout_val_csr(begin), v);
363 break;
364 }
365 begin++;
366 }
367 }
368 ++node2_idx_in_cell;
369 }
370 ++node1_idx_in_cell;
371 }
372 };
373 }
374}
