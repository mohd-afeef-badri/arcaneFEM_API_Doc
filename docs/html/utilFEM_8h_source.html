28/*!
29 * \brief 4th-order tensor used for constitutive operators
30// _ _
31// | D Sup |
32// C = | | with D, S, Sup, Slow = Real3x3 blocks
33// | |
34// | Slow S |
35// |_ _|
36//
37// example: for elastic Tensor4
38// _ _
39// | lambda + 2mu lambda lambda |
40// D = | |
41// | lambda lambda + 2mu lambda |
42// | |
43// | lambda lambda lambda + 2mu |
44// |_ _|
45// _ _
46// | mu 0 0 |
47// S = | | Sup = Slow = 0 (Real3x3)
48// | 0 mu 0 |
49// | |
50// | 0 0 mu |
51// |_ _|
52/*---------------------------------------------------------------------------*/
53/*---------------------------------------------------------------------------*/
54class Tensor4
55{
56 // ***** ATTRIBUTES
57 protected:
58 Real3x3 m_tab[4];//0 = D, 1 = S, 2 = Sup, 3 = Slow
59 bool m_constitutive;// introduced if some special operations are required for constitutive models
60 bool m_sym; // all Tensor2 are symmetric in this case
61
62 public:
63
64 ARCCORE_HOST_DEVICE explicit Tensor4(bool constitutive = true, bool sym = true): m_constitutive(constitutive), m_sym(sym)
65 {}
66
67 ARCCORE_HOST_DEVICE Tensor4(const Real3x3& D,const Real3x3& S){
68 m_constitutive = true;
69 m_tab[0] = D;
70 m_tab[1] = S;
71 m_sym = (real3x3IsSym(D) && real3x3IsSym(S));
72 }
73
74 ARCCORE_HOST_DEVICE Tensor4(const Real3x3& D,const Real3x3& S,const Real3x3& Sup,const Real3x3& Slow,bool constitutive = true) {
75 m_constitutive = constitutive;
76 m_tab[0] = D;
77 m_tab[1] = S;
78 m_tab[2] = Sup;
79 m_tab[3] = Slow;
80 m_sym = (real3x3IsSym(D) && real3x3IsSym(S) && Sup == Slow);
81 }
82
83 ARCCORE_HOST_DEVICE Tensor4(Real lambda, Real mu) {
84 m_sym = true;
85 m_constitutive = true;
86 m_tab[0] = lambda;
87 for (int i = 0; i < 3; i++) m_tab[0][i][i] += 2*mu;
88 m_tab[1] = mu*Real3x3::identity();
89 }
90
91 ARCCORE_HOST_DEVICE Tensor4(const Tensor4& tens) {
92 for (int i = 0; i < 4; i++) m_tab[i] = tens.m_tab[i];
93 m_sym = tens.m_sym;
94 m_constitutive = tens.m_constitutive;
95 }
96
97 ARCCORE_HOST_DEVICE ~Tensor4() = default;
98
99 public:
100
101 ARCCORE_HOST_DEVICE Real3x3 operator[](Int32 i) const {
102 ARCANE_CHECK_AT(i,4);
103 return m_tab[i];
104 }
105 ARCCORE_HOST_DEVICE Real3x3& operator[](Int32 i) {
106 ARCANE_CHECK_AT(i,4);
107 return m_tab[i];
108 }
109
110 //! Return Identity Tensor4
111 static Tensor4 identity() {
112 return { Real3x3::identity(), Real3x3::identity() };
113 }
114
115 //! Return zero Tensor2
116 static Tensor4 zero() {
117 return { Real3x3::zero(), Real3x3::zero() };
118 }
119
120 public:
121 ARCCORE_HOST_DEVICE [[nodiscard]] bool isSymmetric() const { return m_sym; }
122 ARCCORE_HOST_DEVICE void isSymmetric(bool sym) { m_sym = sym; }
123
124 ARCCORE_HOST_DEVICE [[nodiscard]] bool isConstitutive() const { return m_constitutive; }
125 ARCCORE_HOST_DEVICE void isConstitutive(bool b) { m_constitutive = b; }
126
127 ARCCORE_HOST_DEVICE Tensor4& operator=(const Tensor4& tens) {
128 for (int i = 0; i < 4; i++) m_tab[i] = tens.m_tab[i];
129 m_sym = tens.m_sym;
130 m_constitutive = tens.m_constitutive;
131 return (*this);
132 }

222 //! Define the trace of the Tensor
223 ARCCORE_HOST_DEVICE [[nodiscard]] Real trace() const {
224 return real3x3Trace(m_tab[0]) + real3x3Trace(m_tab[1]);

227 //! Get the (i,j) value of the Tensor
228 ARCCORE_HOST_DEVICE [[nodiscard]] Real value_at(Int32 i,Int32 j) const
229 {
230 ARCANE_CHECK_AT(i, 6);
231 ARCANE_CHECK_AT(j, 6);
232
233 if (i < 3) {
234
235 if (j < 3) return m_tab[0][i][j]; // D(i,j)
236 return m_tab[2][i][j-3]; // Sup(i,j-3);
237 }
238
239 if (j < 3) {
240
241 if (m_tab[3] == Real3x3::zero())// if Slow == 0
242 return m_tab[2][j][i-3]; // Sup(j,i-3)
243 return m_tab[3][i-3][j]; // Slow(i-3,j)
244 }
245
246 return m_tab[1][i-3][j-3];// S(i-3,j-3)
247 }
248
249 //! Set the (i,j) value of the Tensor
250 ARCCORE_HOST_DEVICE void value_at(Int32 i,Int32 j, Real x)
251 {
252 ARCANE_CHECK_AT(i, 6);
253 ARCANE_CHECK_AT(j, 6);
254
255 if (i < 3) {
256
257 if (j < 3)
258 m_tab[0][i][j] = x; // D(i,j)
259 else
260 m_tab[2][i][j-3] = x; // Sup(i,j-3);
261
262 } else {
263
264 if (j < 3) {
265
266 if (m_tab[3] == Real3x3::zero()) // if Slow == 0
267 m_tab[2][j][i - 3] = x; // Sup(j,i-3)
268 else
269 m_tab[3][i - 3][j] = x; // Slow(i-3,j)
270
271 } else {
272 m_tab[1][i - 3][j - 3] = x; // S(i-3,j-3)
273 }
274 }
275 }
276
277 //! Get the (i,j,k,l) value of the Tensor4
278 ARCCORE_HOST_DEVICE Real value_at(Int32 i,Int32 j, Int32 k, Int32 l)
279 {
280 ARCANE_CHECK_AT(i, 3);
281 ARCANE_CHECK_AT(j, 3);
282 ARCANE_CHECK_AT(k, 3);
283 ARCANE_CHECK_AT(l, 3);
284
285 int p = math::abs(i - j),q = math::abs(k - l);
286 if (!p)
287 {
288 if (!q) return m_tab[0][i][k]; //D(i,k)
289 if (q == 1 && k) q++;
290 else q--;
291 return m_tab[2][i][q]; //Sup(i,q)
292 }
293
294 if (p == 1 && i) p++;
295 else p--;
296
297 if (!q) {
298
299 if (m_tab[3] == Real3x3::zero()) // if Slow == 0
300 return m_tab[2][p][k];//Sup(p,k)
301
302 return m_tab[3][p][k];//Slow(p,k)
303 }
304 if (q == 1 && k) q++;
305 else q--;
306 return m_tab[1][p][q];//S(p,q)
307 }
308
309 //! Set the (i,j,k,l) value of the Tensor4
310 ARCCORE_HOST_DEVICE void value_at(Int32 i,Int32 j, Int32 k, Int32 l, Real x)
311 {
312 ARCANE_CHECK_AT(i, 3);
313 ARCANE_CHECK_AT(j, 3);
314 ARCANE_CHECK_AT(k, 3);
315 ARCANE_CHECK_AT(l, 3);
316
317 int p = math::abs(i - j),q = math::abs(k - l);
318 if (!p)
319 {
320 if (!q) m_tab[0][i][k] = x;// D(i,k)
321 else
322 {
323 if (q == 1 && k) q++;
324 else q--;
325 m_tab[2][i][q] = x;//Sup(i,q) = x;
326 }
327 }
328 else
329 {
330 if (p == 1 && i) p++;
331 else p--;
332
333 if (!q) {
334
335 if (m_tab[3] == Real3x3::zero()) // if Slow == 0
336 m_tab[2][p][k] = x;//Sup(p,k);
337 else
338 m_tab[3][p][k] = x;//Slow(p,k)
339 }
340 else
341 {
342 if (q == 1 && k) q++;
343 else q--;
344 m_tab[1][p][q] = x;//S(p,q);
345 }
346 }
347 }

349 //! Friend function for scalar multiplication: scalar * Tensor4
350 ARCCORE_HOST_DEVICE friend Tensor4 operator*(Real scalar, const Tensor4& tens) {
351 Tensor4 result;
352 for (Arcane::Int32 i = 0; i < 4; ++i) {
353 result.m_tab[i] = scalar * tens.m_tab[i];
354 }
355 return result;
356 }
