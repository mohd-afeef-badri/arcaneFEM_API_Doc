29void FemModuleElastodynamics::
30startInit()
31{
32 info() << "[ArcaneFem-Info] Started module startInit()";
33 Real elapsedTime = platform::getRealTime();
34
35 m_dofs_on_nodes.initialize(mesh(), mesh()->dimension());
36
37 _getParameters();
38
39 bool use_csr_in_linear_system =
40 options()->linearSystem.serviceName() == "HypreLinearSystem" ||
41 options()->linearSystem.serviceName() == "PETScLinearSystem";
42
43 if (m_matrix_format == "BSR")
44 m_bsr_format.initialize(defaultMesh(), mesh()->dimension(), use_csr_in_linear_system, 0);
45 else if (m_matrix_format == "AF-BSR")
46 m_bsr_format.initialize(defaultMesh(), mesh()->dimension(), use_csr_in_linear_system, 1);
47
48 t = dt;
49 tmax = tmax - dt;
50 m_global_deltat.assign(dt);
51
52 _readCaseTables();
53
54 elapsedTime = platform::getRealTime() - elapsedTime;
55 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(), "initialize", elapsedTime);
56}

58/*---------------------------------------------------------------------------*/
59/*
60 * @brief Computes the FEM simulation for all time step.
61 *
62 * This method performs the main computation of the FEM simulation.
63 * It assembles the linear system, solves it, and updates the variables.
64 * It also handles stopping the computation loop if the maximum time is reached.
65 */
66/*---------------------------------------------------------------------------*/
67
68void FemModuleElastodynamics::
69compute()
70{
71 info() << "[ArcaneFem-Info] Started module compute()";
72 Real elapsedTime = platform::getRealTime();
73
74 // Stop the computation loop if the maximum time is reached
75 if (t >= tmax)
76 subDomain()->timeLoopMng()->stopComputeLoop(true);
77
78 info() << "Time iteration at t : " << t << " (s) ";
79
80 // Set if we want to keep the matrix structure between calls.
81 // The matrix has to have the same structure (same structure for non-zero)
82 bool keep_struct = true;
83 if (m_linear_system.isInitialized() && keep_struct) {
84 m_linear_system.clearValues();
85 }
86 else {
87 m_linear_system.reset();
88 m_linear_system.setLinearSystemFactory(options()->linearSystem());
89 m_linear_system.initialize(subDomain(), acceleratorMng()->defaultRunner(), m_dofs_on_nodes.dofFamily(), "Solver");
90 }
91
92 if (m_petsc_flags != NULL) {
93 CommandLineArguments args = ArcaneFemFunctions::GeneralFunctions::getPetscFlagsFromCommandline(m_petsc_flags);
94 m_linear_system.setSolverCommandLineArguments(args);
95 }
96
97 _doStationarySolve();
98 _updateTime();
99
100 if ((t > tmax + dt - 1e-8) && m_cross_validation)
101 _validateResults();
102
103 elapsedTime = platform::getRealTime() - elapsedTime;
104 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(), "compute", elapsedTime);
105}

133/*
134 * @brief Reads the parameters from the options.
135 *
136 * This method retrieves various parameters such as time step, material properties,
137 * and time discretization methods from the options provided in the simulation.
138 * It also initializes the material properties based on the retrieved parameters.
139 */
140/*---------------------------------------------------------------------------*/
141
142void FemModuleElastodynamics::
143_getParameters()
144{
145 info() << "[ArcaneFem-Info] Started module _getParameters()";
146 Real elapsedTime = platform::getRealTime();
147
148 //--------- time parameters -----------//
149 tmax = options()->tmax(); // max time 𝑡ₘₐₓ
150 dt = options()->dt(); // time step δ𝑡
151
152 //--- damping term parameter ---//
153 etam = options()->etam(); // damping parameter ηₘ
154 etak = options()->etak(); // damping parameter ηₖ
155
156 //--- time discretization parameter ---//
157 alpm = options()->alpm(); // time discretization parameter αᵐ
158 alpf = options()->alpf(); // time discretization parameter ηᶠ
159
160 //--------- material parameter ---------//
161 E = options()->E(); // Youngs modulus 𝐸
162 nu = options()->nu(); // Poisson ratio ν
163 rho = options()->rho(); // Density ρ
164
165 //--------- mesh parameter ------------//
166 m_hex_quad_mesh = options()->hexQuadMesh();
167
168 mu = E / (2 * (1 + nu)); // lame parameter μ
169 lambda = E * nu / ((1 + nu) * (1 - 2 * nu)); // lame parameter λ
170
171 if( options()->mu.isPresent())
172 mu = options()->mu;
173
174 if( options()->lambda.isPresent())
175 lambda = options()->lambda;
176
177 //--------- body force ---------//
178 if (options()->f.isPresent()) {
179 const UniqueArray<String> f_string = options()->f();
180 info() << "[ArcaneFem-Info] Applying Bodyforce " << f_string;
181 for (Int32 i = 0; i < f_string.size(); ++i) {
182 if (f_string[i] != "NULL") {
183 f[i] = std::stod(f_string[i].localstr());
184 }
185 }
186 }
187
188 //----- time discretization Newmark-Beta or Generalized-alpha -----//
189 if (options()->timeDiscretization == "Newmark-beta") {
190
191 info() << "[ArcaneFem-Info] Apply time discretization via Newmark-beta ";
192
193 gamma = 0.5;
194 beta = (1./4.)*(gamma+0.5)*(gamma+0.5) ;
195
196 c0 = rho / (beta * dt * dt) + etam * rho * gamma / beta / dt;
197 c1 = lambda + lambda * etak * gamma / beta / dt;
198 c2 = mu + mu * etak * gamma / beta / dt;
199 c3 = rho / beta / dt - etam * rho * (1 - gamma / beta);
200 c4 = rho * ((1. - 2. * beta) / 2. / beta - etam * dt * (1. - gamma / 2 / beta));
201 c5 = -lambda * etak * gamma / beta / dt;
202 c6 = -mu * etak * gamma / beta / dt;
203 c7 = etak * lambda * (gamma / beta - 1);
204 c8 = etak * lambda * dt * ((1. - 2 * beta) / 2. / beta - (1. - gamma));
205 c9 = etak * mu * (gamma / beta - 1);
206 c10 = etak * mu * dt * ((1. - 2 * beta) / 2. / beta - (1. - gamma));
207 }
208 else if (options()->timeDiscretization == "Generalized-alpha") {
209
210 info() << "[ArcaneFem-Info] Apply time discretization via Generalized-alpha ";
211
212 gamma = 0.5 + alpf - alpm ;
213 beta = (1./4.)*(gamma+0.5)*(gamma+0.5) ;
214
215 c0 = rho * (1. - alpm) / (beta * dt * dt) + etam * rho * gamma * (1 - alpf) / beta / dt;
216 c1 = lambda * (1. - alpf) + lambda * etak * gamma * (1. - alpf) / beta / dt;
217 c2 = mu * (1. - alpf) + mu * etak * gamma * (1. - alpf) / beta / dt;
218 c3 = rho * (1. - alpm) / beta / dt - etam * rho * (1 - gamma * (1 - alpf) / beta);
219 c4 = rho * ((1. - alpm) * (1. - 2. * beta) / 2. / beta - alpm - etam * dt * (1. - alpf) * (1. - gamma / 2 / beta));
220 c5 = lambda * alpf - lambda * etak * gamma * (1. - alpf) / beta / dt;
221 c6 = mu * alpf - mu * etak * gamma * (1. - alpf) / beta / dt;
222 c7 = etak * lambda * (gamma * (1. - alpf) / beta - 1);
223 c8 = etak * lambda * dt * (1. - alpf) * ((1. - 2 * beta) / 2. / beta - (1. - gamma));
224 c9 = etak * mu * (gamma * (1. - alpf) / beta - 1);
225 c10 = etak * mu * dt * (1. - alpf) * ((1. - 2 * beta) / 2. / beta - (1. - gamma));
226 }
227 else {
228 ARCANE_FATAL("Only Newmark-beta | Generalized-alpha are supported for time-discretization ");
229 }
230
231 m_matrix_format = options()->matrixFormat();
232 m_assemble_linear_system = options()->assembleLinearSystem();
233 m_solve_linear_system = options()->solveLinearSystem();
234 m_cross_validation = options()->hasSolutionComparisonFile();
235 m_petsc_flags = options()->petscFlags();
236
237 elapsedTime = platform::getRealTime() - elapsedTime;
238 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(), "get-material-params", elapsedTime);
239}

270void FemModuleElastodynamics::
271_updateVariables()
272{
273 info() << "[ArcaneFem-Info] Started module _updateVariables()";
274 Real elapsedTime = platform::getRealTime();
275
276 VariableDoFReal& dof_u(m_linear_system.solutionVariable());
277 auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
278
279 // First loop: Update displacement values
280 const Int8 dim = mesh()->dimension();
281 ENUMERATE_ (Node, inode, ownNodes()) {
282 Node node = *inode;
283 Real3 u_disp = Real3::zero();
284 for (Int8 i = 0; i < dim; ++i) {
285 u_disp[i] = dof_u[node_dof.dofId(node, i)];
286 }
287 m_dU[node] = u_disp;
288 }
289
290 // Synchronize the variables
291 m_dU.synchronize();
292 m_U.synchronize();
293 m_V.synchronize();
294 m_A.synchronize();
295
296 // Second loop: Update acceleration, velocity and displacement
297 ENUMERATE_ (Node, inode, allNodes()) {
298 Node node = *inode;
299 Real3 aloc = Real3::zero(); // Local acceleration vector
300
301 // Constants for Newmark-beta method
302 const Real beta_dt2 = beta * (dt * dt);
303 const Real beta_factor = (1. - 2. * beta) / (2. * beta);
304 const Real gamma_factor = (1. - gamma);
305
306 // Update each component
307 for (Integer i = 0; i < dim; ++i) {
308 // Calculate new acceleration
309 aloc[i] = (m_dU[node][i] - m_U[node][i] - dt * m_V[node][i]) / beta_dt2 - beta_factor * m_A[node][i];
310
311 // Update velocity, acceleration, and displacement
312 m_V[node][i] += dt * (gamma_factor * m_A[node][i] + gamma * aloc[i]);
313 m_A[node][i] = aloc[i];
314 m_U[node][i] = m_dU[node][i];
315 }
316 }
317
318 elapsedTime = platform::getRealTime() - elapsedTime;
319 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(), "update-variables", elapsedTime);
320}

323/*
324 * @brief Assembles the linear operator for the FEM simulation.
325 *
326 * This method computes the linear operator for the FEM simulation
327 * by assembling the linear parts of the system. It uses the
328 * material properties and the mesh to compute the contributions
329 * from each element. The results are stored in the right-hand side
330 * of the linear system.
331 *
332 * @note This method assumes that the mesh and material properties
333 * are already set up.
334 */
335/*---------------------------------------------------------------------------*/
336
337void FemModuleElastodynamics::
338_assembleLinearOperator()
339{
340 info() << "[ArcaneFem-Info] Started module _assembleLinearOperator()";
341 Real elapsedTime = platform::getRealTime();
342
343 //if(t <= dt) // this works for PETSc but breaks Hypre
344 if(m_matrix_format == "BSR" || m_matrix_format == "AF-BSR")
345 m_bsr_format.toLinearSystem(m_linear_system);
346
347 // Temporary variable to keep values for the RHS part of the linear system
348 VariableDoFReal& rhs_values(m_linear_system.rhsVariable());
349 rhs_values.fill(0.0);
350
351 auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
352
353 _applySourceTerm(rhs_values, node_dof);
354 _applyTraction(rhs_values, node_dof);
355 _applyDirichlet(rhs_values, node_dof);
356
357 elapsedTime = platform::getRealTime() - elapsedTime;
358 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(), "rhs-vector-assembly", elapsedTime);
359}

403/*---------------------------------------------------------------------------*/
404/**
405 * @brief Assembles the FEM bilinear operator on CPU.
406 *
407 * This method assembles the FEM stiffness matrix by iterating over each cell,
408 * computing the element stiffness matrix using the provided function, and
409 * populating the global stiffness matrix accordingly.
410 *
411 * @tparam N Total DOF size (nodes_per_element × dimensions).
412 * @param compute_element_matrix function computing cell's element stiffness matrix.
413 */
414/*---------------------------------------------------------------------------*/
415
416template <int N>
417void FemModuleElastodynamics::
418_assembleBilinearOperatorCpu(const std::function<RealMatrix<N, N>(const Cell&)>& compute_element_matrix)
419{
420 const Int32 dim = mesh()->dimension();
421 auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
422
423 ENUMERATE_ (Cell, icell, allCells()) {
424 Cell cell = *icell;
425 auto K_e = compute_element_matrix(cell);
426
427 Int32 n1_index = 0;
428 for (Node node1 : cell.nodes()) {
429 if (node1.isOwn()) {
430 Int32 n2_index = 0;
431 for (Node node2 : cell.nodes()) {
432 for (Int32 i = 0; i < dim; ++i) {
433 DoFLocalId dof1 = node_dof.dofId(node1, i);
434 for (Int32 j = 0; j < dim; ++j) {
435 DoFLocalId dof2 = node_dof.dofId(node2, j);
436 Real value = K_e(dim * n1_index + i, dim * n2_index + j);
437 m_linear_system.matrixAddValue(dof1, dof2, value);
438 }
439 }
440 ++n2_index;
441 }
442 }
443 ++n1_index;
444 }
445 }
446}

451void FemModuleElastodynamics::
452_assembleBilinearOperator()
453{
454 info() << "[ArcaneFem-Info] Started module _assembleBilinearOperator()";
455 Real elapsedTime = platform::getRealTime();
456
457 if (t <= dt) {
458 if (m_matrix_format == "DOK") {
459 if (mesh()->dimension() == 2) {
460 if (m_hex_quad_mesh) {
461 _assembleBilinearOperatorCpu<8>([this](const Cell& cell) { return _computeElementMatrixQuad4(cell); });
462 }
463 else {
464 _assembleBilinearOperatorCpu<6>([this](const Cell& cell) { return _computeElementMatrixTria3(cell); });
465 }
466 }
467 if (mesh()->dimension() == 3) {
468 if (m_hex_quad_mesh) {
469 _assembleBilinearOperatorCpu<24>([this](const Cell& cell) { return _computeElementMatrixHexa8(cell); });
470 }
471 else{
472 _assembleBilinearOperatorCpu<12>([this](const Cell& cell) { return _computeElementMatrixTetra4(cell); });
473 }
474 }
475 }
476 else if (m_matrix_format == "BSR" || m_matrix_format == "AF-BSR") {
477 if (mesh()->dimension() == 2) {
478 _assembleBilinearOperatorTria3Gpu();
479 }
480 if (mesh()->dimension() == 3) {
481 _assembleBilinearOperatorTetra4Gpu();
482 }
483 }
484 else {
485 ARCANE_FATAL("Unsupported matrix type, only DOK| BSR | AF-BSR is supported.");
486 }
487 }
488
489 elapsedTime = platform::getRealTime() - elapsedTime;
490 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(), "lhs-matrix-assembly", elapsedTime);
491}
