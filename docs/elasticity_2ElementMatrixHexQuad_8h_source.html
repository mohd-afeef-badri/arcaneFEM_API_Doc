15/**
16 * @brief Computes the element matrix for a quadrilateral element (QUAD4, ℙ1 FE).
17 *
18 * Theory:
19 *
20 * a(𝐮,𝐯) = ∫∫ σ(𝐮):ε(𝐯)dΩ with 𝐮 = (𝑢𝑥,𝑢𝑦) and 𝐯 = (𝑣𝑥,𝑣𝑦)
21 * σ(𝐮) is stress tensor with σᵢⱼ = λδᵢⱼεₖₖ + 2μεᵢⱼ
22 * ε(𝐯) is strain tensor with εᵢⱼ = 0.5 (∂𝑣ᵢ/∂xⱼ + ∂𝑣ⱼ/∂xᵢ)
23 *
24 * the bilinear integral expands to
25 *
26 * a(𝐮,𝐯) = ∫∫ [σ_𝑥𝑥ε_𝑥𝑥 + σ_𝑦𝑦ε_𝑦𝑦 + 2σ_𝑥𝑦ε_𝑥𝑦]dΩ
27 *
28 * this further expands to
29 *
30 * a(𝐮,𝐯) = ∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦)
31 * + ∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦)
32 * + ∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥)
33 *
34 * - The first term is "normal strain energy"
35 * - The second term is "compressibility effect"
36 * - The third term is "shear energy"
37 *
38 * Steps involved:
39 * 1. Define Gauss points (2x2) and weights.
40 * 2. Loop over Gauss points to compute the gradients in physical space
41 * and the determinant of the Jacobian, via computeGradientsAndJacobianQuad4.
42 * 3. Compute the integration weight.
43 * 4. Assemble the element matrix using the computed gradients.
44 *
45 * @param cell The cell for which the element matrix is computed.
46 * @return The computed element matrix.
47 */
48/*---------------------------------------------------------------------------*/
49
50ARCCORE_HOST_DEVICE RealMatrix<8, 8> computeElementMatrixQuad4Base(
51const RealVector<4>& dxu, const RealVector<4>& dyu, Real integration_weight, Real lambda, Real mu)
52{
53 // Create displacement gradient vectors for x and y components
54 // For quad4: 4 nodes × 2 DOF = 8 total DOF
55 // Pattern: [u1x, u1y, u2x, u2y, u3x, u3y, u4x, u4y]
56 RealVector<8> dxUx = { dxu(0), 0., dxu(1), 0., dxu(2), 0., dxu(3), 0. };
57 RealVector<8> dyUx = { dyu(0), 0., dyu(1), 0., dyu(2), 0., dyu(3), 0. };
58 RealVector<8> dxUy = { 0., dxu(0), 0., dxu(1), 0., dxu(2), 0., dxu(3) };
59 RealVector<8> dyUy = { 0., dyu(0), 0., dyu(1), 0., dyu(2), 0., dyu(3) };
60
61 // ∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦)
62 RealMatrix<8, 8> normal_strain_energy = (lambda + 2 * mu) * ((dxUx ^ dxUx) + (dyUy ^ dyUy)) * integration_weight;
63
64 // ∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦)
65 RealMatrix<8, 8> compressibility_effect = (lambda) * ((dyUy ^ dxUx) + (dxUx ^ dyUy)) * integration_weight;
66
67 // ∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥)
68 RealMatrix<8, 8> shear_energy = (mu) * ((dxUy + dyUx) ^ (dyUx + dxUy)) * integration_weight;
69
70 return (normal_strain_energy + compressibility_effect + shear_energy);
71}

73RealMatrix<8, 8> FemModuleElasticity::_computeElementMatrixQuad4(Cell cell)
74{
75 // Gauss points and weights for 2x2 quadrature
76 constexpr Real gp[2] = { -M_SQRT1_3, M_SQRT1_3 }; // [-1/sqrt(3) , 1/sqrt(3)]
77 constexpr Real w = 1.0;
78
79 // Initialize the element matrix
80 RealMatrix<8, 8> ae;
81 ae.fill(0.0);
82
83 // Loop over Gauss points
84 for (Int8 ixi = 0; ixi < 2; ++ixi) {
85 for (Int8 ieta = 0; ieta < 2; ++ieta) {
86 // Get the coordinates of the Gauss point in natural coordinates (ξ,η)
87 const Real xi = gp[ixi];
88 const Real eta = gp[ieta];
89
90 // Get shape function gradients w.r.t (𝑥,𝑦) and determinant of Jacobian
91 const auto gp_info = ArcaneFemFunctions::FeOperation2D::computeGradientsAndJacobianQuad4(cell, m_node_coord, xi, eta);
92 const RealVector<4>& dxU = gp_info.dN_dx;
93 const RealVector<4>& dyU = gp_info.dN_dy;
94 const Real detJ = gp_info.det_j;
95
96 // Integration weight
97 const Real integration_weight = detJ * w * w;
98
99 // Add contribution from this Gauss point
100 ae += computeElementMatrixQuad4Base(dxU, dyU, integration_weight, lambda, mu);
101 }
102 }
103
104 return ae;
105}

109 * @brief Computes the element matrix for a quadrilateral element (HEXA8, ℙ1 FE).
110 *
111 * Theory:
112 *
113 * a(𝐮,𝐯) = ∫∫∫ [σ(𝐮):ε(𝐯)dΩ with 𝐮 = (𝑢𝑥, 𝑢𝑦, 𝑢𝑧) and 𝐯 = (𝑣𝑥, 𝑣𝑦, 𝑣𝑧)
114 *
115 * where:
116 *
117 * σ(𝐮) is stress tensor with σᵢⱼ = λδᵢⱼεₖₖ + 2μεᵢⱼ
118 * ε(𝐯) is strain tensor with εᵢⱼ = 0.5 (∂𝑣ᵢ/∂xⱼ + ∂𝑣ⱼ/∂xᵢ)
119 *
120 * the bilinear integral expands to
121 *
122 * a(𝐮,𝐯) = ∫∫∫ [σ_𝑥𝑥ε_𝑥𝑥 + σ_𝑦𝑦ε_𝑦𝑦 + σ_𝑧𝑧ε_𝑧𝑧 + 2σ_𝑥𝑦ε_𝑥𝑦 + 2σ_𝑦𝑧ε_𝑦𝑧 + 2σ_𝑧𝑥ε_𝑧𝑥]dΩ
123 *
124 * this further expands to
125 *
126 * a(𝐮,𝐯) = ∫∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑧/∂𝑧)
127 * + ∫∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑧/∂𝑧 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑧/∂𝑧 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑦/∂𝑦 )
128 * + ∫∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥) + μ(∂𝑢𝑧/∂𝑦 + ∂𝑢𝑦/∂𝑧)(∂𝑣𝑦/∂𝑧 + ∂𝑣𝑧/∂𝑦) + μ(∂𝑢𝑥/∂𝑧 + ∂𝑢𝑧/∂𝑥)(∂𝑣𝑧/∂𝑥 + ∂𝑣𝑥/∂𝑧)
129 *
130 * - The first term is "normal strain energy"
131 * - The second term is "compressibility effect"
132 * - The third term is "shear energy"
133 *
134 */

137ARCCORE_HOST_DEVICE RealMatrix<24, 24> computeElementMatrixHexa8Base(
138const RealVector<8>& dxu, const RealVector<8>& dyu, const RealVector<8>& dzu,
139Real integration_weight, Real lambda, Real mu)
140{
141
142 // Create displacement gradient vectors for x, y, z components
143 // For hex8: 8 nodes × 3 DOF = 24 total DOF
144 // Pattern: [u1x, u1y, u1z, u2x, u2y, u2z, u3x, u3y, u3z, ..., u8x, u8y, u8z]
145
146 // X-displacement gradients
147 RealVector<24> dxUx = { dxu(0), 0., 0., dxu(1), 0., 0., dxu(2), 0., 0., dxu(3), 0., 0.,
148 dxu(4), 0., 0., dxu(5), 0., 0., dxu(6), 0., 0., dxu(7), 0., 0. };
149 RealVector<24> dyUx = { dyu(0), 0., 0., dyu(1), 0., 0., dyu(2), 0., 0., dyu(3), 0., 0.,
150 dyu(4), 0., 0., dyu(5), 0., 0., dyu(6), 0., 0., dyu(7), 0., 0. };
151 RealVector<24> dzUx = { dzu(0), 0., 0., dzu(1), 0., 0., dzu(2), 0., 0., dzu(3), 0., 0.,
152 dzu(4), 0., 0., dzu(5), 0., 0., dzu(6), 0., 0., dzu(7), 0., 0. };
153
154 // Y-displacement gradients
155 RealVector<24> dxUy = { 0., dxu(0), 0., 0., dxu(1), 0., 0., dxu(2), 0., 0., dxu(3), 0.,
156 0., dxu(4), 0., 0., dxu(5), 0., 0., dxu(6), 0., 0., dxu(7), 0. };
157 RealVector<24> dyUy = { 0., dyu(0), 0., 0., dyu(1), 0., 0., dyu(2), 0., 0., dyu(3), 0.,
158 0., dyu(4), 0., 0., dyu(5), 0., 0., dyu(6), 0., 0., dyu(7), 0. };
159 RealVector<24> dzUy = { 0., dzu(0), 0., 0., dzu(1), 0., 0., dzu(2), 0., 0., dzu(3), 0.,
160 0., dzu(4), 0., 0., dzu(5), 0., 0., dzu(6), 0., 0., dzu(7), 0. };
161
162 // Z-displacement gradients
163 RealVector<24> dxUz = { 0., 0., dxu(0), 0., 0., dxu(1), 0., 0., dxu(2), 0., 0., dxu(3),
164 0., 0., dxu(4), 0., 0., dxu(5), 0., 0., dxu(6), 0., 0., dxu(7) };
165 RealVector<24> dyUz = { 0., 0., dyu(0), 0., 0., dyu(1), 0., 0., dyu(2), 0., 0., dyu(3),
166 0., 0., dyu(4), 0., 0., dyu(5), 0., 0., dyu(6), 0., 0., dyu(7) };
167 RealVector<24> dzUz = { 0., 0., dzu(0), 0., 0., dzu(1), 0., 0., dzu(2), 0., 0., dzu(3),
168 0., 0., dzu(4), 0., 0., dzu(5), 0., 0., dzu(6), 0., 0., dzu(7) };
169
170 // Normal strain energy terms: ∫∫∫ (λ+2μ)(∂𝑢𝑥/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑦/∂𝑦 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑧/∂𝑧)
171 RealMatrix<24, 24> normal_strain_energy = (lambda + 2 * mu) *
172 ((dxUx ^ dxUx) + (dyUy ^ dyUy) + (dzUz ^ dzUz)) * integration_weight;
173
174 // Compressibility effects: ∫∫∫ λ(∂𝑢𝑦/∂𝑦 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑦 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑥/∂𝑥 + ∂𝑢𝑥/∂𝑥 ∂𝑣𝑧/∂𝑧 + ∂𝑢𝑦/∂𝑦 ∂𝑣𝑧/∂𝑧 + ∂𝑢𝑧/∂𝑧 ∂𝑣𝑦/∂𝑦 )
175 RealMatrix<24, 24> compressibility_effect = lambda *
176 ((dxUx ^ (dyUy + dzUz)) + (dyUy ^ (dxUx + dzUz)) + (dzUz ^ (dxUx + dyUy))) * integration_weight;
177
178 // Shear energy terms: ∫∫∫ μ(∂𝑢𝑦/∂𝑥 + ∂𝑢𝑥/∂𝑦)(∂𝑣𝑥/∂𝑦 + ∂𝑣𝑦/∂𝑥) +
179 // μ(∂𝑢𝑧/∂𝑦 + ∂𝑢𝑦/∂𝑧)(∂𝑣𝑦/∂𝑧 + ∂𝑣𝑧/∂𝑦) +
180 // μ(∂𝑢𝑥/∂𝑧 + ∂𝑢𝑧/∂𝑥)(∂𝑣𝑧/∂𝑥 + ∂𝑣𝑥/∂𝑧)
181 RealMatrix<24, 24> shear_energy = mu *
182 (((dyUx + dxUy) ^ (dyUx + dxUy)) +
183 ((dzUx + dxUz) ^ (dzUx + dxUz)) +
184 ((dzUy + dyUz) ^ (dzUy + dyUz))) * integration_weight;
185
186 return (normal_strain_energy + compressibility_effect + shear_energy);
187}

189RealMatrix<24, 24> FemModuleElasticity::_computeElementMatrixHexa8(Cell cell)
190{
191 // Gauss points and weights for 2x2x2 quadrature
192 constexpr Real gp[2] = { -M_SQRT1_3, M_SQRT1_3 }; // [-1/sqrt(3), 1/sqrt(3)]
193 constexpr Real w = 1.0;
194
195 // Initialize the element matrix
196 RealMatrix<24, 24> ae;
197 ae.fill(0.0);
198
199 // Loop over Gauss points
200 for (Int8 ixi = 0; ixi < 2; ++ixi) {
201 for (Int8 ieta = 0; ieta < 2; ++ieta) {
202 for (Int8 izeta = 0; izeta < 2; ++izeta) {
203 // Get the coordinates of the Gauss point in natural coordinates (ξ,η,ζ)
204 const Real xi = gp[ixi];
205 const Real eta = gp[ieta];
206 const Real zeta = gp[izeta];
207
208 // Get shape function gradients w.r.t (x,y,z) and determinant of Jacobian
209 const auto gp_info = ArcaneFemFunctions::FeOperation3D::computeGradientsAndJacobianHexa8(
210 cell, m_node_coord, xi, eta, zeta);
211 const RealVector<8>& dxU = gp_info.dN_dx;
212 const RealVector<8>& dyU = gp_info.dN_dy;
213 const RealVector<8>& dzU = gp_info.dN_dz;
214 const Real detJ = gp_info.det_j;
215
216 // Integration weight
217 const Real integration_weight = detJ * w * w * w;
218
219 // Add contribution from this Gauss point
220 ae += computeElementMatrixHexa8Base(dxU, dyU, dzU, integration_weight, lambda, mu);
221 }
222 }
223 }
224
225 return ae;
226}
