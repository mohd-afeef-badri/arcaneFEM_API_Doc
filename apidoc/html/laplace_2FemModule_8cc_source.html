45/*---------------------------------------------------------------------------*/
46/**
47 * @brief Performs the main computation for the FemModuleLaplace.
48 *
49 * This method:
50 * 1. Stops the time loop after 1 iteration since the equation is steady state.
51 * 2. Resets, configures, and initializes the linear system.
52 * 3. Executes the stationary solve.
53 */
54/*---------------------------------------------------------------------------*/
55
56void FemModuleLaplace::
57compute()
58{
59 info() << "[ArcaneFem-Info] Started module compute()";
60 Real elapsedTime = platform::getRealTime();
61
62 // Stop code after computations
63 if (m_global_iteration() > 0)
64 subDomain()->timeLoopMng()->stopComputeLoop(true);
65
66 m_linear_system.reset();
67 m_linear_system.setLinearSystemFactory(options()->linearSystem());
68 m_linear_system.initialize(subDomain(), acceleratorMng()->defaultRunner(), m_dofs_on_nodes.dofFamily(), "Solver");
69
70 if (m_petsc_flags != NULL){
71 CommandLineArguments args = ArcaneFemFunctions::GeneralFunctions::getPetscFlagsFromCommandline(m_petsc_flags);
72 m_linear_system.setSolverCommandLineArguments(args);
73 }
74
75 if (m_matrix_format == "BSR" || m_matrix_format == "AF-BSR") {
76 bool use_csr_in_linear_system =
77 options()->linearSystem.serviceName() == "HypreLinearSystem" ||
78 options()->linearSystem.serviceName() == "PETScLinearSystem";
79 if (m_matrix_format == "BSR")
80 m_bsr_format.initialize(mesh(), 1, use_csr_in_linear_system, 0);
81 else
82 m_bsr_format.initialize(mesh(), 1, use_csr_in_linear_system, 1);
83 m_bsr_format.computeSparsity();
84 }
85
86 _doStationarySolve();
87
88 elapsedTime = platform::getRealTime() - elapsedTime;
89 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(),"compute", elapsedTime);
90}

93/**
94 * @brief Performs a stationary solve for the FEM system.
95 *
96 * This method follows via the following steps:
97 * 1. _assembleBilinearOperator() Assembles the FEM matrix 𝐀
98 * 2. _assembleLinearOperator() Assembles the FEM RHS vector 𝐛
99 * 3. _solve() Solves for solution vector 𝐮 = 𝐀⁻¹𝐛
100 * 4. _updateVariables() Updates FEM variables 𝐮 = 𝐱
101 * 5. _validateResults() Regression test
102 */
103/*---------------------------------------------------------------------------*/
104
105void FemModuleLaplace::
106_doStationarySolve()
107{
108 if(m_assemble_linear_system){
109 _assembleBilinearOperator();
110 _assembleLinearOperator();
111 }
112 if(m_solve_linear_system){
113 _solve();
114 _updateVariables();
115 }
116 if(m_cross_validation){
117 _validateResults();
118 }
119}

123 * @brief Assembles the FEM linear operator for the current simulation step.
124 *
125 * This method constructs the right-hand side (RHS) vector by calling then
126 * appropriate assembly function based on the execution context:
127 * - CPU-exclusive execution: Calls _assembleLinearOperatorCpu().
128 * - CPU or GPU execution: Calls _assembleLinearOperatorGpu().

130 */
131/*---------------------------------------------------------------------------*/
132
133void FemModuleLaplace::_assembleLinearOperator()
134{
135 if (options()->linearSystem.serviceName() == "HypreLinearSystem" ||
136 options()->linearSystem.serviceName() == "PETScLinearSystem")
137 _assembleLinearOperatorGpu();
138 else
139 _assembleLinearOperatorCpu();
140}

142/*---------------------------------------------------------------------------*/
143/**
144 * @brief FEM linear operator for the current simulation step.
145 *
146 * This method constructs the FEM linear systems RHS vector by applying
147 * various boundary conditions and source terms.
148 *
149 * Steps involved:
150 * 1. The RHS vector is initialized to zero before applying any conditions.
151 * 2. If Neumann BC are specified applied to the RHS.
152 * 3. If Dirichlet BC/Point are specified apply to the LHS & RHS.
153 */
154/*---------------------------------------------------------------------------*/
155
156void FemModuleLaplace::
157_assembleLinearOperatorCpu()
158{
159 info() << "[ArcaneFem-Info] Started module _assembleLinearOperator()";
160 Real elapsedTime = platform::getRealTime();
161
162 if (m_matrix_format == "BSR" || m_matrix_format == "AF-BSR")
163 m_bsr_format.toLinearSystem(m_linear_system);
164
165 VariableDoFReal& rhs_values(m_linear_system.rhsVariable()); // Temporary variable to keep values for the RHS
166 rhs_values.fill(0.0);
167
168 auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
169
170 BC::IArcaneFemBC* bc = options()->boundaryConditions();
171 if (bc) {
172 for (BC::INeumannBoundaryCondition* bs : bc->neumannBoundaryConditions()) {
173 if (mesh()->dimension() == 2) {
174 if (m_hex_quad_mesh)
175 ArcaneFemFunctions::BoundaryConditions2D::applyNeumannToRhsQuad4(bs, node_dof, m_node_coord, rhs_values);
176 else
177 ArcaneFemFunctions::BoundaryConditions2D::applyNeumannToRhs(bs, node_dof, m_node_coord, rhs_values);
178 }
179 if (mesh()->dimension() == 3) {
180 if (m_hex_quad_mesh)
181 ArcaneFemFunctions::BoundaryConditions3D::applyNeumannToRhsHexa8(bs, node_dof, m_node_coord, rhs_values);
182 else
183 ArcaneFemFunctions::BoundaryConditions3D::applyNeumannToRhs(bs, node_dof, m_node_coord, rhs_values);
184 }
185 }
186
187 for (BC::IDirichletBoundaryCondition* bs : bc->dirichletBoundaryConditions())
188 ArcaneFemFunctions::BoundaryConditions::applyDirichletToLhsAndRhs(bs, node_dof, m_linear_system, rhs_values);
189
190 for (BC::IDirichletPointCondition* bs : bc->dirichletPointConditions())
191 ArcaneFemFunctions::BoundaryConditions::applyPointDirichletToLhsAndRhs(bs, node_dof, m_linear_system, rhs_values);
192 }
193
194 elapsedTime = platform::getRealTime() - elapsedTime;
195 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(),"rhs-vector-assembly", elapsedTime);
196}
197
198/*---------------------------------------------------------------------------*/
199/**
200 * @brief FEM linear operator for the current simulation step.
201 * GPU compatible. Currently working with HypreDoFLinearSystem.
202 *
203 * This method constructs the linear system by assembling the LHS matrix
204 * and RHS vector, applying various boundary conditions and source terms.
205 *
206 * Steps involved:
207 * 1. The RHS vector is initialized to zero before applying any conditions.
208 * 2. If Neumann BC are specified applied to the RHS.
209 * 3. If Dirichlet BC/Point are specified apply to the LHS & RHS.
210 */
211/*---------------------------------------------------------------------------*/
212
213void FemModuleLaplace::_assembleLinearOperatorGpu()
214{
215 info() << "[ArcaneFem-Info] Started module _assembleLinearOperatorGpu()";
216 Real elapsedTime = platform::getRealTime();
217
218 m_bsr_format.toLinearSystem(m_linear_system);
219
220 auto& rhs_values(m_linear_system.rhsVariable());
221 rhs_values.fill(0.0);
222
223 // Because of Dirichlet (Penalty) implementation in Hypre.
224 m_linear_system.getForcedInfo().fill(false);
225
226 auto queue = subDomain()->acceleratorMng()->defaultQueue();
227 auto mesh_ptr = mesh();
228
229 auto applyBoundaryConditions = [&](auto BCFunctions) {
230 BC::IArcaneFemBC* bc = options()->boundaryConditions();
231
232 if (bc) {
233 for (BC::INeumannBoundaryCondition* bs : bc->neumannBoundaryConditions())
234 BCFunctions.applyNeumannToRhs(bs, m_dofs_on_nodes, m_node_coord, rhs_values, mesh_ptr, queue);
235
236 for (BC::IDirichletBoundaryCondition* bs : bc->dirichletBoundaryConditions())
237 FemUtils::Gpu::BoundaryConditions::applyDirichletToLhsAndRhs(bs, m_dofs_on_nodes, m_linear_system, mesh_ptr, queue);
238
239 for (BC::IDirichletPointCondition* bs : bc->dirichletPointConditions())
240 FemUtils::Gpu::BoundaryConditions::applyPointDirichletToLhsAndRhs(bs, m_dofs_on_nodes, m_linear_system, mesh_ptr, queue);
241 }
242 };
243
244 if (mesh()->dimension() == 3)
245 applyBoundaryConditions(FemUtils::Gpu::BoundaryConditions3D());
246 else
247 applyBoundaryConditions(FemUtils::Gpu::BoundaryConditions2D());
248
249 elapsedTime = platform::getRealTime() - elapsedTime;
250 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(),"rhs-vector-assembly-gpu", elapsedTime);
251}

257/*---------------------------------------------------------------------------*/
258
259void FemModuleLaplace::
260_assembleBilinearOperator()
261{
262 info() << "[ArcaneFem-Info] Started module _assembleBilinearOperator()";
263 Real elapsedTime = platform::getRealTime();
264
265 if (m_matrix_format == "BSR" || m_matrix_format == "AF-BSR") {
266 UnstructuredMeshConnectivityView m_connectivity_view(mesh());
267 auto cn_cv = m_connectivity_view.cellNode();
268 auto m_queue = subDomain()->acceleratorMng()->defaultQueue();
269 auto command = makeCommand(m_queue);
270 auto in_node_coord = ax::viewIn(command, m_node_coord);
271 if (m_matrix_format == "BSR") {
272 if (mesh()->dimension() == 2)
273 m_bsr_format.assembleBilinearAtomic([=] ARCCORE_HOST_DEVICE(CellLocalId cell_lid) { return computeElementMatrixTria3Gpu(cell_lid, cn_cv, in_node_coord); });
274 if (mesh()->dimension() == 3)
275 m_bsr_format.assembleBilinearAtomic([=] ARCCORE_HOST_DEVICE(CellLocalId cell_lid) { return computeElementMatrixTetra4Gpu(cell_lid, cn_cv, in_node_coord); });
276 }
277 if (m_matrix_format == "AF-BSR") {
278 if (mesh()->dimension() == 2)
279 m_bsr_format.assembleBilinearAtomicFree([=] ARCCORE_HOST_DEVICE(CellLocalId cell_lid, Int32 node_lid) { return computeElementVectorTria3Gpu(cell_lid, cn_cv, in_node_coord, node_lid); });
280 if (mesh()->dimension() == 3)
281 m_bsr_format.assembleBilinearAtomicFree([=] ARCCORE_HOST_DEVICE(CellLocalId cell_lid, Int32 node_lid) { return computeElementVectorTetra4Gpu(cell_lid, cn_cv, in_node_coord, node_lid); });
282 }
283 }
284
285 if (m_matrix_format == "DOK") {
286 if (mesh()->dimension() == 3)
287 if(m_hex_quad_mesh)
288 _assembleBilinear<8>([this](const Cell& cell) { return _computeElementMatrixHexa8(cell); });
289 else
290 _assembleBilinear<4>([this](const Cell& cell) { return _computeElementMatrixTetra4(cell); });
291 else
292 if(m_hex_quad_mesh)
293 _assembleBilinear<4>([this](const Cell& cell) { return _computeElementMatrixQuad4(cell); });
294 else
295 _assembleBilinear<3>([this](const Cell& cell) { return _computeElementMatrixTria3(cell); });
296 }
297
298 elapsedTime = platform::getRealTime() - elapsedTime;
299 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(),"lhs-matrix-assembly", elapsedTime);
300}

302/*---------------------------------------------------------------------------*/
303/**
304 * @brief Assembles the bilinear operator matrix for the FEM linear system.
305 *
306 * The method performs the following steps:
307 * 1. Computes element matrix using provided `compute_element_matrix` function.
308 * 2. Assembles global matrix by adding contributions from each cell's element
309 * matrix to the corresponding entries in the global matrix.
310 */
311/*---------------------------------------------------------------------------*/
312
313template <int N>
314void FemModuleLaplace::
315_assembleBilinear(const std::function<RealMatrix<N, N>(const Cell&)>& compute_element_matrix)
316{
317 auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
318
319 ENUMERATE_ (Cell, icell, allCells()) {
320 Cell cell = *icell;
321
322 auto K_e = compute_element_matrix(cell); // element matrix based on the provided function
323 Int32 n1_index = 0;
324 for (Node node1 : cell.nodes()) {
325 Int32 n2_index = 0;
326 for (Node node2 : cell.nodes()) {
327 Real v = K_e(n1_index, n2_index);
328 if (node1.isOwn()) {
329 m_linear_system.matrixAddValue(node_dof.dofId(node1, 0), node_dof.dofId(node2, 0), v);
330 }
331 ++n2_index;
332 }
333 ++n1_index;
334 }
335 }
336}

356/*---------------------------------------------------------------------------*/
357/**
358 * @brief Update the FEM variables.
359 *
360 * This method performs the following actions:
361 * 1. Fetches values of solution from solved linear system to FEM variables,
362 * i.e., it copies RHS DOF to u.
363 * 2. Performs synchronize of FEM variables across subdomains.
364 */
365/*---------------------------------------------------------------------------*/
366
367void FemModuleLaplace::
368_updateVariables()
369{
370 info() << "[ArcaneFem-Module] _updateVariables()";
371 Real elapsedTime = platform::getRealTime();
372
373 {
374 VariableDoFReal& dof_u(m_linear_system.solutionVariable());
375 auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
376 ENUMERATE_ (Node, inode, ownNodes()) {
377 Node node = *inode;
378 m_u[node] = dof_u[node_dof.dofId(node, 0)];
379 }
380 }
381
382 m_u.synchronize();
383
384 elapsedTime = platform::getRealTime() - elapsedTime;
385 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(),"update-variables", elapsedTime);
386}

389/**
390 * @brief Validates and prints the results of the FEM computation.
391 *
392 * This method performs the following actions:
393 * 1. If number of nodes < 200, prints the computed values for each node.
394 * 2. Retrieves the filename for the result file from options.
395 * 3. If a filename is provided, checks the computed results against result file.
396 *
397 * @note The result comparison uses a tolerance of 1.0e-4.
398 */
399/*---------------------------------------------------------------------------*/
400
401void FemModuleLaplace::
402_validateResults()
403{
404 info() << "[ArcaneFem-Info] Started module _validateResults()";
405 Real elapsedTime = platform::getRealTime();
406
407 if (allNodes().size() < 200)
408 ENUMERATE_ (Node, inode, allNodes()) {
409 Node node = *inode;
410 info() << "u[" << node.uniqueId() << "] = " << m_u[node];
411 }
412
413 String filename = options()->solutionComparisonFile();
414 const double epsilon = 1.0e-4;
415 const double min_value_to_test = 1.0e-16;
416
417 checkNodeResultFile(traceMng(), filename, m_u, epsilon, min_value_to_test);
418
419 elapsedTime = platform::getRealTime() - elapsedTime;
420 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(),"result-validation", elapsedTime);
421}
