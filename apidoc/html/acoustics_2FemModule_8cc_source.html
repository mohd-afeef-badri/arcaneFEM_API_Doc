47/**
48 * @brief Performs the main computation for the FemModuleAcoustics.
49 *
50 * This method:
51 * 1. Stops the time loop after 1 iteration since the equation is steady state.
52 * 2. Resets, configures, and initializes the linear system.
53 * 3. Executes the stationary solve.
54 */
55/*---------------------------------------------------------------------------*/
56
57void FemModuleAcoustics::
58compute()
59{
60 info() << "[ArcaneFem-Info] Started module compute()";
61 Real elapsedTime = platform::getRealTime();
62
63 if (m_global_iteration() > 0)
64 subDomain()->timeLoopMng()->stopComputeLoop(true);
65
66 m_linear_system.reset();
67 m_linear_system.setLinearSystemFactory(options()->linearSystem());
68 m_linear_system.initialize(subDomain(), m_dofs_on_nodes.dofFamily(), "Solver");
69
70 if (m_petsc_flags != NULL) {
71 CommandLineArguments args = ArcaneFemFunctions::GeneralFunctions::getPetscFlagsFromCommandline(m_petsc_flags);
72 m_linear_system.setSolverCommandLineArguments(args);
73 }
74
75 _doStationarySolve();
76
77 elapsedTime = platform::getRealTime() - elapsedTime;
78 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(),"compute", elapsedTime);
79}

81/*---------------------------------------------------------------------------*/
82/**
83 * @brief Performs a stationary solve for the FEM system.
84 *
85 * This method follows a sequence of steps to solve FEM system:
86 * 1. _getMaterialParameters() Retrieves material parameters via
87 * 2. _assembleBilinearOperator() Assembles the FEM matrix 𝑨
88 * 3. _assembleLinearOperator() Assembles the FEM RHS vector 𝒃
89 * 4. _solve() Solves for solution vector 𝒖 = 𝑨⁻¹𝒃
90 * 5. _updateVariables() Updates FEM variables 𝒖 = 𝒙
91 * 6. _validateResults() Regression test
92 */
93/*---------------------------------------------------------------------------*/
94
95void FemModuleAcoustics::
96_doStationarySolve()
97{
98 _getMaterialParameters();
99
100 if (m_assemble_linear_system) {
101 _assembleBilinearOperator();
102 _assembleLinearOperator();
103 }
104 if (m_solve_linear_system) {
105 _solve();
106 _updateVariables();
107 }
108 if (m_cross_validation) {
109 _validateResults();
110 }
111}

128/*---------------------------------------------------------------------------*/
129/**
130 * @brief Assembles the FEM linear operator.
131 *
132 * This method assembles the RHS 'b' of the linear system by:
133 * 1. Initializing the RHS values to zero.
134 * 2. Iterating over Neumann boundary conditions to apply it to RHS.
135 */
136/*---------------------------------------------------------------------------*/
137
138void FemModuleAcoustics::
139_assembleLinearOperator()
140{
141 info() << "[ArcaneFem-Info] Started module _assembleLinearOperator()";
142 Real elapsedTime = platform::getRealTime();
143
144 VariableDoFReal& rhs_values(m_linear_system.rhsVariable()); // Temporary variable for RHS values
145 rhs_values.fill(0.0);
146
147 const auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
148
149 BC::IArcaneFemBC* bc = options()->boundaryConditions();
150 if (bc) {
151 for (BC::INeumannBoundaryCondition* bs : bc->neumannBoundaryConditions()) {
152 if (mesh()->dimension() == 2) {
153 if (m_hex_quad_mesh)
154 ArcaneFemFunctions::BoundaryConditions2D::applyNeumannToRhsQuad4(bs, node_dof, m_node_coord, rhs_values);
155 else
156 ArcaneFemFunctions::BoundaryConditions2D::applyNeumannToRhs(bs, node_dof, m_node_coord, rhs_values);
157 }
158 if (mesh()->dimension() == 3) {
159 if (m_hex_quad_mesh)
160 ArcaneFemFunctions::BoundaryConditions3D::applyNeumannToRhsHexa8(bs, node_dof, m_node_coord, rhs_values);
161 else
162 ArcaneFemFunctions::BoundaryConditions3D::applyNeumannToRhs(bs, node_dof, m_node_coord, rhs_values);
163 }
164 }
165 }
166
167 elapsedTime = platform::getRealTime() - elapsedTime;
168 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(), "rhs-vector-assembly", elapsedTime);
169}

189 if (mesh()->dimension() == 2)

194

200/**
201 * @brief Assembles the bilinear operator matrix for the FEM linear system.
202 *
203 * The method performs the following steps:
204 * 1. Computes element matrix using provided `compute_element_matrix` function.
205 * 2. Assembles global matrix by adding contributions from each cell's element
206 * matrix to the corresponding entries in the global matrix.
207 */
208/*---------------------------------------------------------------------------*/
209template <int N>
210void FemModuleAcoustics::
211_assembleBilinear(const std::function<RealMatrix<N, N>(const Cell&)>& compute_element_matrix)
212{
213 auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
214
215 ENUMERATE_ (Cell, icell, allCells()) {
216 Cell cell = *icell;
217
218 auto K_e = compute_element_matrix(cell); // element matrix based on the provided function
219 Int32 n1_index = 0;
220 for (Node node1 : cell.nodes()) {
221 Int32 n2_index = 0;
222 for (Node node2 : cell.nodes()) {
223 Real v = K_e(n1_index, n2_index);
224 if (node1.isOwn())
225 m_linear_system.matrixAddValue(node_dof.dofId(node1, 0), node_dof.dofId(node2, 0), v);
226 ++n2_index;
227 }
228 ++n1_index;
229 }
230 }
231}

256/*---------------------------------------------------------------------------*/
257/**
258 * @brief Update the FEM variables.
259 *
260 * This method performs the following actions:
261 * 1. Fetches values of solution from solved linear system to FEM variables,
262 * i.e., it copies RHS DOF to u.
263 * 2. Performs synchronize of FEM variables across subdomains.
264 */
265/*---------------------------------------------------------------------------*/
266
267void FemModuleAcoustics::
268_updateVariables()
269{
270 info() << "[ArcaneFem-Module] _updateVariables()";
271 Real elapsedTime = platform::getRealTime();
272
273 {
274 VariableDoFReal& dof_u(m_linear_system.solutionVariable());
275 auto node_dof(m_dofs_on_nodes.nodeDoFConnectivityView());
276 ENUMERATE_ (Node, inode, ownNodes()) {
277 Node node = *inode;
278 m_u[node] = dof_u[node_dof.dofId(node, 0)];
279 }
280 }
281
282 m_u.synchronize();
283
284 elapsedTime = platform::getRealTime() - elapsedTime;
285 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(), "update-variables", elapsedTime);
286}

289/**
290 * @brief Validates and prints the results of the FEM computation.
291 *
292 * This method performs the following actions:
293 * 1. If number of nodes < 200, prints the computed values for each node.
294 * 2. Retrieves the filename for the solution-comparison-file from options.

296 *
297 * @note The result comparison uses a tolerance of 1.0e-4.
298 */
299/*---------------------------------------------------------------------------*/
300
301void FemModuleAcoustics::
302_validateResults()
303{
304 info() << "[ArcaneFem-Info] Started module _validateResults()";
305 Real elapsedTime = platform::getRealTime();
306
307 if (allNodes().size() < 200)
308 ENUMERATE_ (Node, inode, allNodes()) {
309 Node node = *inode;
310 info() << "u[" << node.uniqueId() << "] = " << m_u[node];
311 }
312
313 String filename = options()->solutionComparisonFile();
314 const double epsilon = 1.0e-4;
315 const double min_value_to_test = 1.0e-16;
316
317 checkNodeResultFile(traceMng(), filename, m_u, epsilon, min_value_to_test);
318
319 elapsedTime = platform::getRealTime() - elapsedTime;
320 ArcaneFemFunctions::GeneralFunctions::printArcaneFemTime(traceMng(), "result-validation", elapsedTime);
321}
