26inline void FemModuleElasticity::
27_applyDirichlet(VariableDoFReal& rhs_values, const IndexedNodeDoFConnectivityView& node_dof)
28{
29 // check if Hypre|PETSc solver is used and delegate to GPU for dirichlet assembly
30 auto use_gpu = options()->linearSystem.serviceName() == "HypreLinearSystem" ||
31 options()->linearSystem.serviceName() == "PETScLinearSystem";
32 if (use_gpu) {
33 _assembleDirichletsGpu();
34 return;
35 }
36
37 info() << "[ArcaneFem-Info] Started module _assembleLinearOperatorCpu()";
38
39 BC::IArcaneFemBC* bc = options()->boundaryConditions();
40 if (bc) {
41 for (BC::IDirichletBoundaryCondition* bs : bc->dirichletBoundaryConditions())
42 ArcaneFemFunctions::BoundaryConditions::applyDirichletToLhsAndRhs(bs, node_dof, m_linear_system, rhs_values);
43
44 for (BC::IDirichletPointCondition* bs : bc->dirichletPointConditions())
45 ArcaneFemFunctions::BoundaryConditions::applyPointDirichletToLhsAndRhs(bs, node_dof, m_linear_system, rhs_values);
46 }
47}

59/*---------------------------------------------------------------------------*/
60
61void FemModuleElasticity::_assembleDirichletsGpu()
62{
63 info() << "[ArcaneFem-Info] Started module _assembleLinearOperatorGpu()";
64
65 auto queue = subDomain()->acceleratorMng()->defaultQueue();
66 auto mesh_ptr = mesh();
67
68 BC::IArcaneFemBC* bc = options()->boundaryConditions();
69
70 if (bc) {
71 for (BC::IDirichletBoundaryCondition* bs : bc->dirichletBoundaryConditions())
72 FemUtils::Gpu::BoundaryConditions::applyDirichletToLhsAndRhs(bs, m_dofs_on_nodes, m_linear_system, mesh_ptr, queue);
73
74 for (BC::IDirichletPointCondition* bs : bc->dirichletPointConditions())
75 FemUtils::Gpu::BoundaryConditions::applyPointDirichletToLhsAndRhs(bs, m_dofs_on_nodes, m_linear_system, mesh_ptr, queue);
76 }
77}
