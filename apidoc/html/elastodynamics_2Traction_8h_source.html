15/**
16 * @brief Applies traction to the RHS vector of the linear system.
17 *
18 * This function computes the contribution of traction to the RHS vector
19 * of the linear system. It iterates over all cells in the mesh, calculates
20 * the appropriate force contributions based on the element type and mesh
21 * dimension, and updates the RHS vector accordingly.
22 *
23 * traction term ∫∫ (𝐭.𝐯) with 𝐭 = (𝑡𝑥, 𝑡𝑦, 𝑡𝑧) = (t[0], t[1], t[2])
24 *
25 * @param rhs_values The variable representing the RHS vector to be updated.
26 * @param node_dof The connectivity view mapping nodes to their corresponding
27 * degrees of freedom (DoFs).
28 *
29/*---------------------------------------------------------------------------*/
30
31inline void FemModuleElastodynamics::
32_applyTraction(VariableDoFReal& rhs_values, const IndexedNodeDoFConnectivityView& node_dof)
33{
34 Int32 boundary_condition_index = 0;
35 BC::IArcaneFemBC* bc = options()->boundaryConditions();
36
37 for (BC::ITractionBoundaryCondition* bs : bc->tractionBoundaryConditions()) {
38 const auto traction_table_file_name = bs->getTractionInputFile();
39 const bool is_transient_traction = !traction_table_file_name.empty();
40
41 auto transientTraction = [&](auto fn) { fn(bs, t, boundary_condition_index, m_traction_case_table_list, node_dof, m_node_coord, rhs_values); };
42 auto constantTraction = [&](auto fn) { fn(bs, node_dof, m_node_coord, rhs_values); };
43
44 if (mesh()->dimension() == 2) {
45 if (m_hex_quad_mesh)
46 is_transient_traction ? transientTraction(ArcaneFemFunctions::BoundaryConditions2D::applyTractionTableToRhsQuad4)
47 : constantTraction(ArcaneFemFunctions::BoundaryConditions2D::applyTractionToRhsQuad4);
48 else
49 is_transient_traction ? transientTraction(ArcaneFemFunctions::BoundaryConditions2D::applyTractionTableToRhsTria3)
50 : constantTraction(ArcaneFemFunctions::BoundaryConditions2D::applyTractionToRhsTria3);
51 }
52 else if (mesh()->dimension() == 3) {
53 if (m_hex_quad_mesh)
54 is_transient_traction ? transientTraction(ArcaneFemFunctions::BoundaryConditions3D::applyTractionTableToRhsHexa8)
55 : constantTraction(ArcaneFemFunctions::BoundaryConditions3D::applyTractionToRhsHexa8);
56 else
57 is_transient_traction ? transientTraction(ArcaneFemFunctions::BoundaryConditions3D::applyTractionTableToRhsTetra4)
58 : constantTraction(ArcaneFemFunctions::BoundaryConditions3D::applyTractionToRhsTetra4);
59 }
60 }
61}
