15/**
16 * @brief Computes the element matrix for a triangular element (ℙ1 FE).
17 *
18 * This function calculates the integral of the expression:
19 * a(𝐮,𝐯) = ∫∫ (∂𝑢𝑥/∂𝑥 ∂𝑣𝑦/∂𝑥 + ∂𝑢𝑦/∂𝑥 ∂𝑣𝑦/∂𝑦)dΩ
20 * + ∫∫ (∂𝑢𝑦/∂𝑥 ∂𝑣𝑥/∂𝑥 + ∂𝑣𝑦/∂𝑦 ∂𝑣𝑥/∂𝑦)dΩ
21 * + ∫∫ (𝑢𝑦 𝑣𝑦)dΩ
22 *
23 * Steps involved:
24 * 1. Calculate the area of the triangle.
25 * 2. Compute the gradients of the shape functions.
26 * 3. Return a(𝑢,𝑣);
27 */
28/*---------------------------------------------------------------------------*/
29
30RealMatrix<6, 6> FemModuleBilaplacian::
31_computeElementMatrixTria3(Cell cell)
32{
33 Real3 dxu = ArcaneFemFunctions::FeOperation2D::computeGradientXTria3(cell, m_node_coord);
34 Real3 dyu = ArcaneFemFunctions::FeOperation2D::computeGradientYTria3(cell, m_node_coord);
35 Real area = ArcaneFemFunctions::MeshOperation::computeAreaTria3(cell, m_node_coord);
36
37 RealVector<6> Uy = { 0., 1., 0., 1., 0., 1. };
38 RealVector<6> Ux = { 1., 0., 1., 0., 1., 0. };
39
40 RealVector<6> dxUx = { dxu[0], 0., dxu[1], 0., dxu[2], 0. };
41 RealVector<6> dyUx = { dyu[0], 0., dyu[1], 0., dyu[2], 0. };
42 RealVector<6> dxUy = { 0., dxu[0], 0., dxu[1], 0., dxu[2] };
43 RealVector<6> dyUy = { 0., dyu[0], 0., dyu[1], 0., dyu[2] };
44
45 RealMatrix<6, 6> int_Omega_i = ((dxUx ^ dxUy) + (dyUx ^ dyUy)) * area + ((dxUy ^ dxUx) + (dyUy ^ dyUx)) * area + massMatrix(Uy, Uy) * area;
46
47 return int_Omega_i;
48}
